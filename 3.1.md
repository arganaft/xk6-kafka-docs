echo "🔧 Repackage RisingWave в минимальный Alpine + JRE 17"

def ApplicationName       = "risingwave"
def RISINGWAVE_VERSION    = "v2.7.3"
def NexusGroupId          = 'ci01142241'
def NexusArtifactId       = "ci02631387/${ApplicationName}"
def dockerRegistryURL     = 'docker-dev.registry-ci.delta.sbrf.ru'

sh "mkdir -p docker_build/risingwave"
dir("docker_build/risingwave") {

    // ========== 1. Предварительная проверка базовых образов ==========
    echo "=" * 60
    echo "📥 ШАГ 1: Проверяем доступность базовых образов"
    echo "=" * 60

    echo "📥 1.1: Пуллим официальный образ RisingWave..."
    sh """
        docker pull sberosc.sigma.sbrf.ru/docker.io/risingwavelabs/risingwave:${RISINGWAVE_VERSION} && \\
            echo '✅ RisingWave образ скачан успешно' || \\
            echo '❌ ОШИБКА: Не удалось скачать образ RisingWave!'
    """

    echo "📥 1.2: Пуллим Alpine..."
    sh """
        docker pull sberosc.sigma.sbrf.ru/docker.io/alpine:3.23.3 && \\
            echo '✅ Alpine образ скачан успешно' || \\
            echo '❌ ОШИБКА: Не удалось скачать Alpine!'
    """

    echo "📥 1.3: Размеры скачанных образов:"
    sh "docker images | grep -E 'risingwave|alpine' || echo '❌ Образы не найдены в локальном кэше!'"

    // ========== 2. Разведка: что внутри официального образа ==========
    echo "=" * 60
    echo "🔍 ШАГ 2: Исследуем содержимое официального образа RisingWave"
    echo "=" * 60

    sh """
        echo '--- 2.1: Корень /risingwave ---'
        docker run --rm --entrypoint sh \\
            sberosc.sigma.sbrf.ru/docker.io/risingwavelabs/risingwave:${RISINGWAVE_VERSION} \\
            -c 'ls -la /risingwave/ 2>/dev/null || echo "❌ /risingwave не существует"'

        echo ''
        echo '--- 2.2: Бинарники /risingwave/bin ---'
        docker run --rm --entrypoint sh \\
            sberosc.sigma.sbrf.ru/docker.io/risingwavelabs/risingwave:${RISINGWAVE_VERSION} \\
            -c 'ls -lah /risingwave/bin/ 2>/dev/null || echo "❌ /risingwave/bin не существует"'

        echo ''
        echo '--- 2.3: Библиотеки /risingwave/lib ---'
        docker run --rm --entrypoint sh \\
            sberosc.sigma.sbrf.ru/docker.io/risingwavelabs/risingwave:${RISINGWAVE_VERSION} \\
            -c 'ls -lah /risingwave/lib/ 2>/dev/null || echo "❌ /risingwave/lib не существует"'

        echo ''
        echo '--- 2.4: JAR-файлы (Java-коннекторы) ---'
        docker run --rm --entrypoint sh \\
            sberosc.sigma.sbrf.ru/docker.io/risingwavelabs/risingwave:${RISINGWAVE_VERSION} \\
            -c 'find /risingwave -name "*.jar" -exec ls -lh {} \\; 2>/dev/null || echo "❌ JAR файлы не найдены"'

        echo ''
        echo '--- 2.5: Java внутри образа ---'
        docker run --rm --entrypoint sh \\
            sberosc.sigma.sbrf.ru/docker.io/risingwavelabs/risingwave:${RISINGWAVE_VERSION} \\
            -c 'java -version 2>&1 || echo "❌ Java не установлена в образе"'

        echo ''
        echo '--- 2.6: Путь к Java (JAVA_HOME) ---'
        docker run --rm --entrypoint sh \\
            sberosc.sigma.sbrf.ru/docker.io/risingwavelabs/risingwave:${RISINGWAVE_VERSION} \\
            -c 'find / -name "java" -type f 2>/dev/null | head -10 || echo "❌ java binary не найден"'

        echo ''
        echo '--- 2.7: Размер /risingwave целиком ---'
        docker run --rm --entrypoint sh \\
            sberosc.sigma.sbrf.ru/docker.io/risingwavelabs/risingwave:${RISINGWAVE_VERSION} \\
            -c 'du -sh /risingwave/ 2>/dev/null || echo "❌ Ошибка"'

        echo ''
        echo '--- 2.8: Версия RisingWave ---'
        docker run --rm \\
            sberosc.sigma.sbrf.ru/docker.io/risingwavelabs/risingwave:${RISINGWAVE_VERSION} \\
            --version 2>&1 || echo '❌ risingwave --version не работает'

        echo ''
        echo '--- 2.9: Какая ОС в базовом образе ---'
        docker run --rm --entrypoint sh \\
            sberosc.sigma.sbrf.ru/docker.io/risingwavelabs/risingwave:${RISINGWAVE_VERSION} \\
            -c 'cat /etc/os-release 2>/dev/null || echo "❌ Не определена"'

        echo ''
        echo '--- 2.10: ldd бинарника (какие .so нужны) ---'
        docker run --rm --entrypoint sh \\
            sberosc.sigma.sbrf.ru/docker.io/risingwavelabs/risingwave:${RISINGWAVE_VERSION} \\
            -c 'ldd /risingwave/bin/risingwave 2>/dev/null || echo "❌ ldd не работает"'
    """

    // ========== 3. Генерируем Dockerfile ==========
    echo "=" * 60
    echo "📝 ШАГ 3: Создаём Dockerfile"
    echo "=" * 60

    writeFile file: 'Dockerfile', text: """\
########################################################
# Stage 1: Извлекаем всё из официального образа
########################################################
FROM sberosc.sigma.sbrf.ru/docker.io/risingwavelabs/risingwave:${RISINGWAVE_VERSION} AS source

# --- Логируем что доступно ---
RUN echo '========== STAGE 1: Содержимое source ==========' && \\
    echo '--- /risingwave ---' && ls -la /risingwave/ && \\
    echo '--- /risingwave/bin ---' && ls -lah /risingwave/bin/ && \\
    echo '--- Размер ---' && du -sh /risingwave/ && \\
    echo '========== STAGE 1: OK =========='

########################################################
# Stage 2: Чистый Alpine + gcompat + JRE 17
########################################################
FROM sberosc.sigma.sbrf.ru/docker.io/alpine:3.23.3

RUN echo '========== STAGE 2: Начало =========='

# --- 2.1: Корпоративное зеркало APK ---
RUN sed -i 's|https://dl-cdn.alpinelinux.org/alpine/|https://token:1234567890@sberosc.sigma.sbrf.ru/repo/apk/|g' /etc/apk/repositories && \\
    echo '✅ 2.1: APK repositories переключены на корпоративное зеркало' && \\
    cat /etc/apk/repositories

# --- 2.2: Установка runtime-зависимостей ---
RUN echo '⏳ 2.2: Устанавливаем runtime-пакеты...' && \\
    apk add --no-cache --no-check-certificate \\
        ca-certificates \\
        gcompat \\
        libgcc \\
        libstdc++ \\
        openssl \\
        zlib \\
        lz4-libs \\
        zstd-libs \\
        snappy && \\
    echo '✅ 2.2: Runtime-пакеты установлены' && \\
    echo '--- Установленные пакеты ---' && apk list --installed 2>/dev/null | wc -l && \\
    echo 'пакетов всего'

# --- 2.3: Установка JRE 17 ---
RUN echo '⏳ 2.3: Устанавливаем OpenJDK 17 JRE...' && \\
    apk add --no-cache --no-check-certificate \\
        openjdk17-jre-headless && \\
    echo '✅ 2.3: JRE 17 установлен' && \\
    java -version 2>&1

# --- 2.4: Копируем бинарник RisingWave ---
COPY --from=source /risingwave/bin/risingwave /usr/bin/risingwave
RUN echo '⏳ 2.4: Проверяем бинарник risingwave...' && \\
    ls -lh /usr/bin/risingwave && \\
    file /usr/bin/risingwave && \\
    echo '✅ 2.4: Бинарник скопирован' && \\
    risingwave --version 2>&1 && \\
    echo '✅ 2.4: Бинарник запускается!' || \\
    echo '⚠️ 2.4: Бинарник скопирован, но --version не отработал (проверим позже)'

# --- 2.5: Копируем Java-коннекторы и библиотеки ---
COPY --from=source /risingwave/ /risingwave/
RUN echo '⏳ 2.5: Проверяем скопированные файлы...' && \\
    echo '--- /risingwave ---' && ls -la /risingwave/ && \\
    echo '--- /risingwave/bin ---' && ls -lah /risingwave/bin/ 2>/dev/null || true && \\
    echo '--- JAR файлы ---' && find /risingwave -name '*.jar' -exec ls -lh {} \\; 2>/dev/null || echo 'Нет JAR файлов' && \\
    echo '--- Общий размер /risingwave ---' && du -sh /risingwave/ && \\
    echo '✅ 2.5: Файлы скопированы'

# --- 2.6: Переменные окружения ---
ENV JAVA_HOME=/usr/lib/jvm/java-17-openjdk
ENV PATH="\${JAVA_HOME}/bin:\${PATH}"
ENV RW_CONNECTOR_RPC_ENDPOINT=0.0.0.0:50051
RUN echo '✅ 2.6: ENV установлены' && \\
    echo "JAVA_HOME=\${JAVA_HOME}" && \\
    echo "PATH=\${PATH}" && \\
    which java && java -version 2>&1

# --- 2.7: Безопасность: non-root пользователь ---
RUN adduser -D -s /sbin/nologin rwuser && \\
    echo '✅ 2.7: Пользователь rwuser создан' && \\
    id rwuser

# --- 2.8: Каталоги для данных ---
RUN mkdir -p /data /risingwave/state_store && \\
    chown -R rwuser:rwuser /data /risingwave && \\
    echo '✅ 2.8: Каталоги созданы и права назначены' && \\
    ls -la /data && ls -la /risingwave/

# --- 2.9: Удаляем лишнее ---
RUN echo '⏳ 2.9: Удаляем ненужное...' && \\
    echo '--- До очистки ---' && df -h / && \\
    rm -rf /usr/share/doc /usr/share/man /var/cache/apk/* && \\
    echo '--- После очистки ---' && df -h / && \\
    echo '✅ 2.9: Очистка завершена'

# --- 2.10: Финальная проверка ---
RUN echo '========================================' && \\
    echo '   ФИНАЛЬНАЯ ПРОВЕРКА ОБРАЗА' && \\
    echo '========================================' && \\
    echo '' && \\
    echo '--- ОС ---' && cat /etc/os-release | head -3 && \\
    echo '' && \\
    echo '--- Бинарник ---' && ls -lh /usr/bin/risingwave && \\
    echo '' && \\
    echo '--- Java ---' && java -version 2>&1 && \\
    echo '' && \\
    echo '--- Все установленные пакеты ---' && apk list --installed 2>/dev/null && \\
    echo '' && \\
    echo '--- Количество пакетов ---' && apk list --installed 2>/dev/null | wc -l && \\
    echo '' && \\
    echo '--- Размер /risingwave ---' && du -sh /risingwave/ && \\
    echo '--- Размер /usr ---' && du -sh /usr/ && \\
    echo '--- Общий размер файловой системы ---' && du -sh / 2>/dev/null || true && \\
    echo '' && \\
    echo '========== ✅ ОБРАЗ СОБРАН УСПЕШНО =========='

USER rwuser
WORKDIR /home/rwuser

EXPOSE 4566 5691 5690 50051

ENTRYPOINT ["risingwave"]
CMD ["single_node"]
""".stripIndent()

    sh "cat -n Dockerfile"
    sh "echo 'Строк в Dockerfile:' && wc -l Dockerfile"

    // ========== 4. Чистим старое ==========
    echo "=" * 60
    echo "🧹 ШАГ 4: Удаляем предыдущий образ (если был)"
    echo "=" * 60
    sh "docker rmi ${dockerRegistryURL}/${NexusGroupId}/${NexusArtifactId}:${RISINGWAVE_VERSION} 2>/dev/null || echo 'Предыдущего образа не было — ОК'"

    // ========== 5. Собираем ==========
    echo "=" * 60
    echo "⏳ ШАГ 5: Собираем образ (ожидаемое время: 3-5 минут)"
    echo "=" * 60

    def startTime = System.currentTimeMillis()

    sh """
        DOCKER_BUILDKIT=0 docker build \\
            --pull \\
            --no-cache \\
            --progress=plain \\
            -t ${dockerRegistryURL}/${NexusGroupId}/${NexusArtifactId}:${RISINGWAVE_VERSION} \\
            . 2>&1 | tee build.log
    """

    def duration = (System.currentTimeMillis() - startTime) / 1000
    echo "⏱️ Сборка заняла: ${duration} секунд"

    // ========== 6. Проверяем результат ==========
    echo "=" * 60
    echo "🔍 ШАГ 6: Проверяем собранный образ"
    echo "=" * 60

    echo "6.1: Размер образа:"
    sh "docker images | grep risingwave || echo '❌ ОБРАЗ НЕ НАЙДЕН!'"

    echo "6.2: История слоёв образа:"
    sh "docker history ${dockerRegistryURL}/${NexusGroupId}/${NexusArtifactId}:${RISINGWAVE_VERSION} || echo '❌ Не удалось получить историю'"

    echo "6.3: risingwave --version:"
    sh """
        docker run --rm \\
            ${dockerRegistryURL}/${NexusGroupId}/${NexusArtifactId}:${RISINGWAVE_VERSION} \\
            --version 2>&1 && echo '✅ Бинарник работает!' || echo '❌ Бинарник НЕ работает!'
    """

    echo "6.4: Java внутри нового образа:"
    sh """
        docker run --rm --entrypoint sh \\
            ${dockerRegistryURL}/${NexusGroupId}/${NexusArtifactId}:${RISINGWAVE_VERSION} \\
            -c 'java -version 2>&1' && echo '✅ Java работает!' || echo '❌ Java НЕ работает!'
    """

    echo "6.5: Список пакетов в новом образе:"
    sh """
        docker run --rm --entrypoint sh \\
            ${dockerRegistryURL}/${NexusGroupId}/${NexusArtifactId}:${RISINGWAVE_VERSION} \\
            -c 'apk list --installed 2>/dev/null | sort' || echo '⚠️ Не удалось получить список'
    """

    // ========== 7. Пушим ==========
    echo "=" * 60
    echo "📤 ШАГ 7: Пушим образ в Nexus"
    echo "=" * 60

    sh """
        docker image push ${dockerRegistryURL}/${NexusGroupId}/${NexusArtifactId}:${RISINGWAVE_VERSION} && \\
            echo '✅ Образ запушен успешно!' || \\
            echo '❌ ОШИБКА при пуше!'
    """

    echo "=" * 60
    echo "✅ ГОТОВО: ${ApplicationName}:${RISINGWAVE_VERSION}"
    echo "=" * 60
}
