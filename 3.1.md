echo "🔧 Сборка минимального образа RisingWave"

def ApplicationName       = "risingwave"
def RISINGWAVE_VERSION    = "v2.7.3"
def NexusGroupId          = 'ci01142241'
def NexusArtifactId       = "ci02631387/${ApplicationName}"
def dockerRegistryURL     = 'docker-dev.registry-ci.delta.sbrf.ru'

sh "mkdir -p docker_build/risingwave"
dir("docker_build/risingwave") {

    echo "📝 Создаём Dockerfile (multi-stage repackage)"
    writeFile file: 'Dockerfile', text: """
    ########################################################
    # Stage 1: Источник — официальный образ RisingWave
    # Используем ТОЛЬКО для извлечения скомпилированного
    # бинарника. Этот слой НЕ попадёт в итоговый образ.
    ########################################################
    FROM sberosc.sigma.sbrf.ru/docker.io/risingwavelabs/risingwave:${RISINGWAVE_VERSION} AS source

    ########################################################
    # Stage 2: Минимальный рантайм на базе debian-slim
    # (glibc-совместим с бинарником из Ubuntu-образа)
    # debian:bookworm-slim ~75MB, против ~600MB+ Ubuntu
    ########################################################
    FROM sberosc.sigma.sbrf.ru/docker.io/debian:bookworm-slim AS runtime

    # ---- Минимальные runtime-зависимости ----
    RUN apt-get update && \\
        apt-get install -y --no-install-recommends \\
            ca-certificates \\
            libssl3 \\
            libgcc-s1 \\
            libstdc++6 \\
            liblzma5 \\
            zlib1g \\
        && rm -rf /var/lib/apt/lists/* \\
        && apt-get clean \\
        && rm -rf /var/cache/apt/* \\
        && rm -rf /tmp/* /var/tmp/*

    # ---- Копируем ТОЛЬКО бинарник RisingWave ----
    COPY --from=source /risingwave/bin/risingwave /usr/local/bin/risingwave

    # ---- Если нужны коннекторы (Kafka и др.) — раскомментировать: ----
    # COPY --from=source /risingwave/bin/connector-node /usr/local/bin/connector-node
    # COPY --from=source /risingwave/lib/ /usr/local/lib/risingwave/

    # ---- Безопасность: non-root пользователь ----
    RUN groupadd -r rwuser && useradd -r -g rwuser -m -s /usr/sbin/nologin rwuser

    # ---- Каталог для данных ----
    RUN mkdir -p /data /risingwave && \\
        chown -R rwuser:rwuser /data /risingwave

    # ---- Удаляем всё лишнее для уменьшения attack surface ----
    RUN rm -rf /usr/share/doc /usr/share/man /usr/share/info /usr/share/locale && \\
        rm -f /usr/bin/apt* /usr/bin/dpkg* 2>/dev/null || true

    USER rwuser
    WORKDIR /home/rwuser

    EXPOSE 4566 5691

    ENTRYPOINT ["risingwave"]
    CMD ["single_node"]
    """

    sh "cat Dockerfile"

    echo "🏗️ Собираем очищенный образ"
    sh "docker build . -t ${dockerRegistryURL}/${NexusGroupId}/${NexusArtifactId}:${RISINGWAVE_VERSION}"

    echo "🔍 Проверяем что бинарник работает"
    sh "docker run --rm ${dockerRegistryURL}/${NexusGroupId}/${NexusArtifactId}:${RISINGWAVE_VERSION} --version || echo 'WARNING: version check skipped'"

    echo "📤 Пушим образ в Nexus"
    sh "docker image push ${dockerRegistryURL}/${NexusGroupId}/${NexusArtifactId}:${RISINGWAVE_VERSION}"

    echo "✅ Образ ${ApplicationName}:${RISINGWAVE_VERSION} успешно собран и запушен"
}
