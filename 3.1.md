Ниже полный класс `V2StreamViewConfigToSqlConverter`.

Он ожидает, что до него уже отработал `V2ViewConfigValidator`, то есть на вход приходит `V2ValidatedViewConfig`, где каждый `fieldKey` уже сопоставлен с реальной колонкой ClickHouse.

Ключевая логика:

```text
fieldKey:    FxSpotForwardTrade.baseCurrency.code
columnName:  baseCurrency_code
alias:       FxSpotForwardTrade.baseCurrency.code
```

SQL будет:

```sql
"baseCurrency_code" AS "FxSpotForwardTrade.baseCurrency.code"
```

---

```java
package com.pivotService.SQLBuilder.v2;

import com.fasterxml.jackson.databind.JsonNode;
import com.fasterxml.jackson.databind.node.ArrayNode;
import com.fasterxml.jackson.databind.node.JsonNodeFactory;
import com.pivotService.SQLBuilder.v2.dialect.V2ClickHouseDialect;
import com.pivotService.SQLBuilder.v2.validation.V2SqlFieldRef;
import com.pivotService.SQLBuilder.v2.validation.V2ValidatedViewConfig;
import org.springframework.stereotype.Service;

import java.util.*;
import java.util.stream.Collectors;

@Service
public class V2StreamViewConfigToSqlConverter {

    private final V2ClickHouseDialect dialect;

    public V2StreamViewConfigToSqlConverter(V2ClickHouseDialect dialect) {
        this.dialect = dialect;
    }

    public String convert(V2ValidatedViewConfig validated) {
        if (validated == null) {
            throw new IllegalArgumentException("Validated config cannot be null");
        }

        JsonNode config = validated.config();

        if (config == null || !config.isObject()) {
            throw new IllegalArgumentException("Config must be JSON object");
        }

        ArrayNode columns = getArrayNode(config, "columns");
        ArrayNode rowGroupColumns = getArrayNode(config, "rowGroupColumns");
        ArrayNode pivotColumns = getArrayNode(config, "pivotColumns");
        ArrayNode valueColumns = getArrayNode(config, "valueColumns");
        ArrayNode sortModel = getArrayNode(config, "sortModel");
        ArrayNode groupKeys = getArrayNode(config, "groupKeys");

        JsonNode filterModel = config.path("filterModel");

        int startRow = config.path("startRow").asInt(0);
        int endRow = config.path("endRow").asInt(100);

        columns = ensureIdColumn(validated, columns);

        Map<String, String> aggregationByField = extractAggregationByField(valueColumns);

        boolean hasGrouping = rowGroupColumns != null && !rowGroupColumns.isEmpty();

        ArrayNode cleanGroupColumns = cleanGroupColumns(rowGroupColumns, groupKeys);
        ArrayNode firstRowGroupColumns = firstGroupLevel(cleanGroupColumns, pivotColumns);

        Set<String> actualGroupByFields = extractFieldKeySet(firstRowGroupColumns);

        StringBuilder sql = new StringBuilder();

        sql.append("SELECT ");
        sql.append(buildSelectClause(
                validated,
                columns,
                firstRowGroupColumns,
                valueColumns,
                hasGrouping
        ));

        sql.append(" FROM ");
        sql.append(dialect.quoteTable(validated.tableName()));

        FilterResult filterResult = buildFilters(
                validated,
                filterModel,
                groupKeys,
                rowGroupColumns,
                aggregationByField
        );

        if (!filterResult.whereClause.isEmpty()) {
            sql.append(" WHERE ").append(filterResult.whereClause);
        }

        if (!firstRowGroupColumns.isEmpty()) {
            String groupByClause = buildGroupByClause(
                    validated,
                    firstRowGroupColumns,
                    hasGrouping
            );

            if (!groupByClause.isEmpty()) {
                sql.append(" GROUP BY ").append(groupByClause);
            }
        }

        if (!filterResult.havingClause.isEmpty()) {
            sql.append(" HAVING ").append(filterResult.havingClause);
        }

        String orderByClause = buildOrderByClause(
                validated,
                sortModel,
                actualGroupByFields,
                valueColumns
        );

        if (!orderByClause.isEmpty()) {
            sql.append(" ORDER BY ").append(orderByClause);
        }

        int limit = endRow - startRow;

        if (limit > 0) {
            sql.append(" LIMIT ").append(limit);
        }

        if (startRow > 0) {
            sql.append(" OFFSET ").append(startRow);
        }

        return sql.toString();
    }

    // =============================================================================================
    // SELECT
    // =============================================================================================

    private String buildSelectClause(
            V2ValidatedViewConfig validated,
            ArrayNode columns,
            ArrayNode firstRowGroupColumns,
            ArrayNode valueColumns,
            boolean hasGrouping
    ) {
        List<String> selectParts = new ArrayList<>();

        if (!firstRowGroupColumns.isEmpty()) {
            for (JsonNode groupColumn : firstRowGroupColumns) {
                selectParts.add(groupExpression(validated, groupColumn, true));
            }

            addAggregations(validated, selectParts, valueColumns);

            return selectParts.isEmpty() ? "*" : String.join(", ", selectParts);
        }

        if (hasGrouping) {
            addColumnsToSelect(validated, selectParts, columns);

            return selectParts.isEmpty() ? "*" : String.join(", ", selectParts);
        }

        addColumnsToSelect(validated, selectParts, columns);

        return selectParts.isEmpty() ? "*" : String.join(", ", selectParts);
    }

    private void addColumnsToSelect(
            V2ValidatedViewConfig validated,
            List<String> selectParts,
            ArrayNode columns
    ) {
        if (columns == null || columns.isEmpty()) {
            selectParts.add("*");
            return;
        }

        Set<String> addedAliases = new LinkedHashSet<>();

        for (JsonNode columnNode : columns) {
            String fieldKey = getFieldKey(columnNode);

            if (fieldKey == null || fieldKey.isBlank()) {
                continue;
            }

            V2SqlFieldRef field = validated.field(fieldKey);

            String alias = field.alias();

            if (!addedAliases.add(alias)) {
                continue;
            }

            selectParts.add(
                    columnExpression(field) +
                            " AS " +
                            aliasExpression(field)
            );
        }
    }

    private void addAggregations(
            V2ValidatedViewConfig validated,
            List<String> selectParts,
            ArrayNode valueColumns
    ) {
        if (valueColumns == null || valueColumns.isEmpty()) {
            return;
        }

        Set<String> addedAliases = new LinkedHashSet<>();

        for (JsonNode valueColumn : valueColumns) {
            String fieldKey = getFieldKey(valueColumn);
            String aggFunc = getAggFunc(valueColumn);

            if (fieldKey == null || aggFunc == null) {
                continue;
            }

            V2SqlFieldRef field = validated.field(fieldKey);

            String alias = field.alias();

            if (!addedAliases.add(alias)) {
                continue;
            }

            String expression = dialect.aggregationExpression(
                    aggFunc,
                    columnExpression(field)
            );

            selectParts.add(expression + " AS " + aliasExpression(field));
        }
    }

    // =============================================================================================
    // WHERE / HAVING
    // =============================================================================================

    private FilterResult buildFilters(
            V2ValidatedViewConfig validated,
            JsonNode filterModel,
            ArrayNode groupKeys,
            ArrayNode rowGroupColumns,
            Map<String, String> aggregationByField
    ) {
        FilterResult result = new FilterResult();

        List<String> whereConditions = new ArrayList<>();
        List<String> havingConditions = new ArrayList<>();

        if (filterModel != null && !filterModel.isMissingNode() && filterModel.isObject()) {
            Iterator<Map.Entry<String, JsonNode>> fields = filterModel.fields();

            while (fields.hasNext()) {
                Map.Entry<String, JsonNode> entry = fields.next();

                String fieldKey = entry.getKey();
                JsonNode filterDef = entry.getValue();

                V2SqlFieldRef field = validated.field(fieldKey);

                boolean isAggregated = aggregationByField.containsKey(fieldKey);

                String expression;

                if (isAggregated) {
                    expression = dialect.aggregationExpression(
                            aggregationByField.get(fieldKey),
                            columnExpression(field)
                    );
                } else {
                    expression = columnExpression(field);
                }

                String filterSql = buildFilterCondition(
                        field,
                        expression,
                        filterDef
                );

                if (filterSql == null || filterSql.isBlank()) {
                    continue;
                }

                if (isAggregated) {
                    havingConditions.add(filterSql);
                } else {
                    whereConditions.add(filterSql);
                }
            }
        }

        addGroupKeyConditions(
                validated,
                groupKeys,
                rowGroupColumns,
                whereConditions
        );

        result.whereClause = String.join(" AND ", whereConditions);
        result.havingClause = String.join(" AND ", havingConditions);

        return result;
    }

    private void addGroupKeyConditions(
            V2ValidatedViewConfig validated,
            ArrayNode groupKeys,
            ArrayNode rowGroupColumns,
            List<String> whereConditions
    ) {
        if (groupKeys == null || groupKeys.isEmpty()) {
            return;
        }

        Map<String, String> groupByByField = extractGroupByByField(rowGroupColumns);

        for (JsonNode node : groupKeys) {
            if (!node.hasNonNull("fieldKey") || !node.has("value")) {
                continue;
            }

            String fieldKey = node.get("fieldKey").asText();
            JsonNode valueNode = node.get("value");

            V2SqlFieldRef field = validated.field(fieldKey);

            String expression = columnExpression(field);

            String groupBy = node.path("groupBy").asText(null);

            if (groupBy == null || groupBy.isBlank()) {
                groupBy = groupByByField.get(fieldKey);
            }

            String literal;

            if (groupBy != null && !groupBy.isBlank()) {
                expression = dialect.dateGroupExpression(
                        expression,
                        groupBy,
                        field.kind()
                );

                literal = dialect.dateGroupLiteral(
                        valueNode.asText(),
                        groupBy
                );
            } else {
                literal = dialect.literal(
                        valueNode.isNull() ? null : valueNode.asText(),
                        field.kind()
                );
            }

            whereConditions.add(expression + " = " + literal);
        }
    }

    private String buildFilterCondition(
            V2SqlFieldRef field,
            String expression,
            JsonNode filterDef
    ) {
        if (filterDef == null || filterDef.isMissingNode() || filterDef.isNull()) {
            return null;
        }

        String filterType = filterDef.path("filterType").asText("");

        if (filterDef.has("operator")) {
            String operator = filterDef.path("operator").asText("AND");

            String cond1 = buildSingleFilterCondition(
                    field,
                    expression,
                    filterDef.get("condition1"),
                    filterType
            );

            String cond2 = buildSingleFilterCondition(
                    field,
                    expression,
                    filterDef.get("condition2"),
                    filterType
            );

            List<String> parts = new ArrayList<>();

            if (cond1 != null && !cond1.isBlank()) {
                parts.add(cond1);
            }

            if (cond2 != null && !cond2.isBlank()) {
                parts.add(cond2);
            }

            if (parts.isEmpty()) {
                return null;
            }

            if (parts.size() == 1) {
                return parts.get(0);
            }

            String sqlOperator = "OR".equalsIgnoreCase(operator)
                    ? " OR "
                    : " AND ";

            return "(" + String.join(sqlOperator, parts) + ")";
        }

        if ("set".equalsIgnoreCase(filterType)) {
            return buildSetFilter(field, expression, filterDef);
        }

        return buildSingleFilterCondition(
                field,
                expression,
                filterDef,
                filterType
        );
    }

    private String buildSingleFilterCondition(
            V2SqlFieldRef field,
            String expression,
            JsonNode condition,
            String filterType
    ) {
        if (condition == null || condition.isMissingNode() || condition.isNull()) {
            return null;
        }

        String type = condition.path("type").asText("");

        return switch (type) {
            case "equals" -> {
                String value = extractFilterValue(condition, filterType, false);

                yield value == null
                        ? null
                        : expression + " = " + literal(field, value);
            }

            case "notEqual" -> {
                String value = extractFilterValue(condition, filterType, false);

                yield value == null
                        ? null
                        : expression + " != " + literal(field, value);
            }

            case "contains" -> expression +
                    " LIKE " +
                    dialect.stringLiteral("%" + condition.path("filter").asText("") + "%");

            case "notContains" -> expression +
                    " NOT LIKE " +
                    dialect.stringLiteral("%" + condition.path("filter").asText("") + "%");

            case "startsWith" -> expression +
                    " LIKE " +
                    dialect.stringLiteral(condition.path("filter").asText("") + "%");

            case "endsWith" -> expression +
                    " LIKE " +
                    dialect.stringLiteral("%" + condition.path("filter").asText(""));

            case "lessThan" -> {
                String value = extractFilterValue(condition, filterType, false);

                yield value == null
                        ? null
                        : expression + " < " + literal(field, value);
            }

            case "lessThanOrEqual" -> {
                String value = extractFilterValue(condition, filterType, false);

                yield value == null
                        ? null
                        : expression + " <= " + literal(field, value);
            }

            case "greaterThan" -> {
                String value = extractFilterValue(condition, filterType, false);

                yield value == null
                        ? null
                        : expression + " > " + literal(field, value);
            }

            case "greaterThanOrEqual" -> {
                String value = extractFilterValue(condition, filterType, false);

                yield value == null
                        ? null
                        : expression + " >= " + literal(field, value);
            }

            case "inRange" -> {
                String from = extractFilterValue(condition, filterType, false);
                String to = extractFilterValue(condition, filterType, true);

                if (from == null || to == null) {
                    yield null;
                }

                yield expression +
                        " BETWEEN " +
                        literal(field, from) +
                        " AND " +
                        literal(field, to);
            }

            case "blank", "empty" -> "(" + expression + " IS NULL)";

            case "notBlank", "notEmpty" -> "(" + expression + " IS NOT NULL)";

            default -> {
                if (condition.has("filter")) {
                    String value = extractFilterValue(condition, filterType, false);

                    yield value == null
                            ? null
                            : expression + " = " + literal(field, value);
                }

                yield null;
            }
        };
    }

    private String buildSetFilter(
            V2SqlFieldRef field,
            String expression,
            JsonNode filterDef
    ) {
        JsonNode values = filterDef.path("values");

        if (!values.isArray() || values.isEmpty()) {
            return expression + " IN (NULL)";
        }

        StringJoiner joiner = new StringJoiner(", ");
        boolean hasNull = false;

        for (JsonNode value : values) {
            if (value == null || value.isNull()) {
                hasNull = true;
            } else {
                joiner.add(dialect.literal(value.asText(), field.kind()));
            }
        }

        if (hasNull && joiner.length() > 0) {
            return "(" +
                    expression +
                    " IN (" +
                    joiner +
                    ") OR " +
                    expression +
                    " IS NULL)";
        }

        if (hasNull) {
            return expression + " IS NULL";
        }

        return expression + " IN (" + joiner + ")";
    }

    // =============================================================================================
    // GROUP BY
    // =============================================================================================

    private String buildGroupByClause(
            V2ValidatedViewConfig validated,
            ArrayNode rowGroupColumns,
            boolean hasGrouping
    ) {
        if (!hasGrouping) {
            return "";
        }

        if (rowGroupColumns == null || rowGroupColumns.isEmpty()) {
            return "";
        }

        List<String> groupByParts = new ArrayList<>();

        for (JsonNode node : rowGroupColumns) {
            groupByParts.add(groupExpression(validated, node, false));
        }

        return String.join(", ", groupByParts);
    }

    private String groupExpression(
            V2ValidatedViewConfig validated,
            JsonNode node,
            boolean addAlias
    ) {
        String fieldKey = node.path("fieldKey").asText(null);

        if (fieldKey == null || fieldKey.isBlank()) {
            throw new IllegalArgumentException("group fieldKey is required");
        }

        V2SqlFieldRef field = validated.field(fieldKey);

        String expression = columnExpression(field);

        String groupBy = node.path("groupBy").asText(null);

        if (groupBy != null && !groupBy.isBlank()) {
            expression = dialect.dateGroupExpression(
                    expression,
                    groupBy,
                    field.kind()
            );
        }

        if (addAlias) {
            return expression + " AS " + aliasExpression(field);
        }

        return expression;
    }

    // =============================================================================================
    // ORDER BY
    // =============================================================================================

    private String buildOrderByClause(
            V2ValidatedViewConfig validated,
            ArrayNode sortModel,
            Set<String> actualGroupByFields,
            ArrayNode valueColumns
    ) {
        if (sortModel == null || sortModel.isEmpty()) {
            return "";
        }

        Set<String> valueFields = valueColumns != null
                ? valueColumns.findValuesAsText("fieldKey")
                        .stream()
                        .collect(Collectors.toCollection(LinkedHashSet::new))
                : Set.of();

        List<String> orderParts = new ArrayList<>();

        for (JsonNode sortItem : sortModel) {
            String colId = sortItem.path("colId").asText(null);

            if (colId == null || colId.isBlank()) {
                continue;
            }

            if (!isAllowedOrderBy(colId, actualGroupByFields, valueFields)) {
                continue;
            }

            V2SqlFieldRef field = validated.field(colId);

            String sort = sortItem.path("sort").asText("asc");

            String expression;

            if (actualGroupByFields.contains(colId) || valueFields.contains(colId)) {
                expression = aliasExpression(field);
            } else {
                expression = columnExpression(field);
            }

            orderParts.add(
                    expression +
                            " " +
                            ("desc".equalsIgnoreCase(sort) ? "DESC" : "ASC")
            );
        }

        return String.join(", ", orderParts);
    }

    private boolean isAllowedOrderBy(
            String colId,
            Set<String> groupByFields,
            Set<String> valueFields
    ) {
        if (groupByFields.isEmpty() && valueFields.isEmpty()) {
            return true;
        }

        return groupByFields.contains(colId) || valueFields.contains(colId);
    }

    // =============================================================================================
    // Expressions
    // =============================================================================================

    private String columnExpression(V2SqlFieldRef field) {
        return dialect.quoteColumn(field.columnName());
    }

    private String aliasExpression(V2SqlFieldRef field) {
        return dialect.quoteAlias(field.alias());
    }

    private String literal(V2SqlFieldRef field, String value) {
        return dialect.literal(value, field.kind());
    }

    // =============================================================================================
    // Array preparation
    // =============================================================================================

    private ArrayNode ensureIdColumn(
            V2ValidatedViewConfig validated,
            ArrayNode columns
    ) {
        ArrayNode result = JsonNodeFactory.instance.arrayNode();

        boolean hasPhysicalIdColumn = false;

        if (columns != null) {
            for (JsonNode column : columns) {
                result.add(column.deepCopy());

                String fieldKey = getFieldKey(column);

                if (fieldKey == null || fieldKey.isBlank()) {
                    continue;
                }

                V2SqlFieldRef field = validated.field(fieldKey);

                if ("id".equals(field.columnName())) {
                    hasPhysicalIdColumn = true;
                }
            }
        }

        if (!hasPhysicalIdColumn) {
            var idColumn = JsonNodeFactory.instance.objectNode();
            idColumn.put("fieldKey", "id");
            result.add(idColumn);
        }

        return result;
    }

    private ArrayNode cleanGroupColumns(
            ArrayNode rowGroupColumns,
            ArrayNode groupKeys
    ) {
        ArrayNode result = JsonNodeFactory.instance.arrayNode();

        if (rowGroupColumns == null || rowGroupColumns.isEmpty()) {
            return result;
        }

        Set<String> rowGroupFieldKeys = extractFieldKeySet(rowGroupColumns);
        Set<String> groupKeySet = extractFieldKeySet(groupKeys);

        rowGroupFieldKeys.removeAll(groupKeySet);

        for (JsonNode column : rowGroupColumns) {
            String fieldKey = getFieldKey(column);

            if (fieldKey != null && rowGroupFieldKeys.contains(fieldKey)) {
                result.add(column.deepCopy());
            }
        }

        return result;
    }

    private ArrayNode firstGroupLevel(
            ArrayNode cleanGroupColumns,
            ArrayNode pivotColumns
    ) {
        ArrayNode result = JsonNodeFactory.instance.arrayNode();

        if (cleanGroupColumns != null && !cleanGroupColumns.isEmpty()) {
            result.add(cleanGroupColumns.get(0).deepCopy());
        }

        if (pivotColumns != null && !pivotColumns.isEmpty()) {
            for (JsonNode pivotColumn : pivotColumns) {
                result.add(pivotColumn.deepCopy());
            }
        }

        return result;
    }

    // =============================================================================================
    // Extractors
    // =============================================================================================

    private Map<String, String> extractAggregationByField(ArrayNode valueColumns) {
        Map<String, String> result = new LinkedHashMap<>();

        if (valueColumns == null || valueColumns.isEmpty()) {
            return result;
        }

        for (JsonNode valueColumn : valueColumns) {
            String fieldKey = getFieldKey(valueColumn);
            String aggFunc = getAggFunc(valueColumn);

            if (fieldKey != null && aggFunc != null) {
                result.put(fieldKey, aggFunc);
            }
        }

        return result;
    }

    private Map<String, String> extractGroupByByField(ArrayNode rowGroupColumns) {
        Map<String, String> result = new LinkedHashMap<>();

        if (rowGroupColumns == null || rowGroupColumns.isEmpty()) {
            return result;
        }

        for (JsonNode node : rowGroupColumns) {
            String fieldKey = node.path("fieldKey").asText(null);
            String groupBy = node.path("groupBy").asText(null);

            if (fieldKey != null && groupBy != null && !groupBy.isBlank()) {
                result.put(fieldKey, groupBy);
            }
        }

        return result;
    }

    private Set<String> extractFieldKeySet(ArrayNode arrayNode) {
        Set<String> result = new LinkedHashSet<>();

        if (arrayNode == null || arrayNode.isEmpty()) {
            return result;
        }

        for (JsonNode node : arrayNode) {
            String fieldKey = getFieldKey(node);

            if (fieldKey != null && !fieldKey.isBlank()) {
                result.add(fieldKey);
            }
        }

        return result;
    }

    private String extractFilterValue(
            JsonNode condition,
            String filterType,
            boolean secondValue
    ) {
        if (condition == null || condition.isMissingNode() || condition.isNull()) {
            return null;
        }

        if ("date".equalsIgnoreCase(filterType)) {
            if (secondValue) {
                if (condition.hasNonNull("dateTo")) {
                    return condition.get("dateTo").asText();
                }

                if (condition.hasNonNull("filterTo")) {
                    return condition.get("filterTo").asText();
                }

                return null;
            }

            if (condition.hasNonNull("dateFrom")) {
                return condition.get("dateFrom").asText();
            }
        }

        String fieldName = secondValue ? "filterTo" : "filter";

        if (!condition.hasNonNull(fieldName)) {
            return null;
        }

        return condition.get(fieldName).asText();
    }

    private String getFieldKey(JsonNode column) {
        if (column == null || column.isMissingNode() || column.isNull()) {
            return null;
        }

        return column.path("fieldKey").asText(null);
    }

    private String getAggFunc(JsonNode column) {
        if (column == null || column.isMissingNode() || column.isNull()) {
            return null;
        }

        String aggFunc = column.path("aggFunc").asText(null);

        if (aggFunc == null || aggFunc.isBlank()) {
            return null;
        }

        return switch (aggFunc.toLowerCase(Locale.ROOT)) {
            case "sum", "avg", "min", "max", "count", "unique_count" -> aggFunc.toLowerCase(Locale.ROOT);
            default -> null;
        };
    }

    private ArrayNode getArrayNode(JsonNode parent, String fieldName) {
        if (parent == null || parent.isMissingNode() || parent.isNull()) {
            return null;
        }

        JsonNode node = parent.path(fieldName);

        return node.isArray() ? (ArrayNode) node : null;
    }

    private static class FilterResult {
        private String whereClause = "";
        private String havingClause = "";
    }
}
```

---

Для примера, если в `v2/config` пришло:

```json
{
  "fieldKey": "FxSpotForwardTrade.baseCurrency.code"
}
```

а resolver определил:

```text
columnName = baseCurrency_code
alias = FxSpotForwardTrade.baseCurrency.code
```

то этот converter сгенерирует:

```sql
"baseCurrency_code" AS "FxSpotForwardTrade.baseCurrency.code"
```
