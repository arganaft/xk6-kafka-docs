    private void validateFilterValueIfPossible(
            V2SqlFieldRef field,
            JsonNode filterDef,
            String path,
            ValidationErrors errors
    ) {
        if (filterDef == null || filterDef.isMissingNode() || filterDef.isNull()) {
            return;
        }

        if (filterDef.has("operator")) {
            validateFilterValueIfPossible(field, filterDef.path("condition1"), path + ".condition1", errors);
            validateFilterValueIfPossible(field, filterDef.path("condition2"), path + ".condition2", errors);
            return;
        }

        String type = filterDef.path("type").asText("");
        String filterType = filterDef.path("filterType").asText("");

        if ("set".equalsIgnoreCase(filterType)) {
            JsonNode values = filterDef.path("values");

            if (values.isArray()) {
                for (int i = 0; i < values.size(); i++) {
                    if (!values.get(i).isNull()) {
                        fieldValueValidator.validateLiteral(
                                field.metadata(),
                                values.get(i).asText(),
                                path + ".values[" + i + "]",
                                errors
                        );
                    }
                }
            }

            return;
        }

        switch (type) {
            case "equals", "notEqual", "lessThan", "lessThanOrEqual",
                 "greaterThan", "greaterThanOrEqual" -> {
                String value = extractFilterValue(filterDef, filterType, false);

                if (value != null) {
                    fieldValueValidator.validateLiteral(field.metadata(), value, path, errors);
                }
            }

            case "inRange" -> {
                String from = extractFilterValue(filterDef, filterType, false);
                String to = extractFilterValue(filterDef, filterType, true);

                if (from != null) {
                    fieldValueValidator.validateLiteral(field.metadata(), from, path + ".from", errors);
                }

                if (to != null) {
                    fieldValueValidator.validateLiteral(field.metadata(), to, path + ".to", errors);
                }
            }

            case "contains", "notContains", "startsWith", "endsWith" -> {
                if (!field.kind().isTextLike()) {
                    errors.reject(
                            path + ".type",
                            "INVALID_FILTER_OPERATION",
                            "Text filter operation is not allowed for field type " + field.kind()
                    );
                }

                String value = filterDef.path("filter").asText(null);

                if (value != null) {
                    fieldValueValidator.validateLiteral(field.metadata(), value, path + ".filter", errors);
                }
            }

            case "blank", "notBlank", "empty", "notEmpty", "" -> {
                // value not required
            }

            default -> errors.reject(
                    path + ".type",
                    "UNKNOWN_FILTER_OPERATION",
                    "Unknown filter operation: " + type
            );
        }
    }

    private String extractFilterValue(JsonNode condition, String filterType, boolean secondValue) {
        if ("date".equalsIgnoreCase(filterType)) {
            if (secondValue) {
                if (condition.hasNonNull("dateTo")) {
                    return condition.get("dateTo").asText();
                }

                if (condition.hasNonNull("filterTo")) {
                    return condition.get("filterTo").asText();
                }

                return null;
            }

            if (condition.hasNonNull("dateFrom")) {
                return condition.get("dateFrom").asText();
            }
        }

        String fieldName = secondValue ? "filterTo" : "filter";

        if (!condition.hasNonNull(fieldName)) {
            return null;
        }

        return condition.get(fieldName).asText();
    }

    private V2SqlFieldRef resolveField(
            V2LiveBookSchema schema,
            String fieldKey,
            String path,
            Map<String, V2SqlFieldRef> fields,
            ValidationErrors errors
    ) {
        if (fieldKey == null || fieldKey.isBlank()) {
            errors.reject(path, "FIELD_KEY_EMPTY", "fieldKey must not be empty");
            return null;
        }

        V2SqlFieldRef existing = fields.get(fieldKey);

        if (existing != null) {
            return existing;
        }

        if ("id".equals(fieldKey)) {
            V2SqlFieldRef idRef = technicalIdRef();
            fields.put(fieldKey, idRef);
            return idRef;
        }

        V2ResolvedLiveBookField liveBookField = schema.findField(fieldKey.toLowerCase()).orElse(null);

        if (liveBookField == null) {
            errors.reject(
                    path,
                    "UNKNOWN_LIVEBOOK_FIELD",
                    "Unknown fieldKey '" + fieldKey + "' for LiveBook '" +
                            schema.liveBookName() + "'. v2/config expects full field path from LiveBook.fields."
            );
            return null;
        }

        V2SqlFieldRef ref = new V2SqlFieldRef(
                fieldKey,
                liveBookField.columnName(),
                liveBookField.alias(),
                liveBookField.kind(),
                liveBookField.fieldMetadata()
        );

        fields.put(fieldKey, ref);

        return ref;
    }

    private void addTechnicalIdField(
            Map<String, V2SqlFieldRef> fields,
            V2LiveBookSchema schema
    ) {
        boolean alreadyHasIdColumn = fields.values()
                .stream()
                .anyMatch(field -> "id".equals(field.columnName()));

        if (alreadyHasIdColumn) {
            return;
        }

        fields.put("id", technicalIdRef());
    }

    private V2SqlFieldRef technicalIdRef() {
        BusinessField idField = new BusinessField(
                "id",
                "LONG",
                "id",
                List.of()
        );

        FieldMetadata metadata = new FieldMetadata(
                "Entity",
                idField,
                FieldKind.LONG
        );

        return new V2SqlFieldRef(
                "Entity.id",
                "id",
                "id",
                FieldKind.LONG,
                metadata
        );
    }

    private void validateGroupByIfPresent(
            JsonNode item,
            V2SqlFieldRef field,
            String path,
            ValidationErrors errors
    ) {
        if (field == null || !item.hasNonNull("groupBy")) {
            return;
        }

        String groupBy = item.get("groupBy").asText().toLowerCase(Locale.ROOT);

        if (!Set.of("year", "month", "day").contains(groupBy)) {
            errors.reject(
                    path + ".groupBy",
                    "INVALID_GROUP_BY",
                    "groupBy must be one of year, month, day"
            );
            return;
        }

        if (!field.kind().isDateLike()) {
            errors.reject(
                    path + ".groupBy",
                    "GROUP_BY_REQUIRES_DATE",
                    "Date groupBy requires date-like field, but got " + field.kind()
            );
        }
    }

    private void validateAggFunc(
            V2SqlFieldRef field,
            String aggFunc,
            String path,
            ValidationErrors errors
    ) {
        if (aggFunc == null || aggFunc.isBlank()) {
            errors.reject(path, "AGG_FUNC_REQUIRED", "aggFunc is required");
            return;
        }

        String normalized = aggFunc.toLowerCase(Locale.ROOT);

        if (!AGG_FUNCS.contains(normalized)) {
            errors.reject(path, "INVALID_AGG_FUNC", "aggFunc must be one of " + AGG_FUNCS);
            return;
        }

        if (field == null) {
            return;
        }

        if (Set.of("sum", "avg").contains(normalized) && !field.kind().isNumeric()) {
            errors.reject(
                    path,
                    "AGG_FUNC_REQUIRES_NUMERIC",
                    "Aggregation '" + normalized + "' requires numeric field, but got " + field.kind()
            );
        }
    }

    private void ensureEntityIdColumnPresent(JsonNode config, V2LiveBookSchema schema) {
        // 1. Проверяем, что такое поле вообще существует в LiveBook
        if (schema.findField("entity.id").isEmpty()) {
            return; // если нет, ничего не добавляем, чтобы не плодить ошибки валидации
        }

        JsonNode columns = config.path("columns");
        if (columns.isArray()) {
            for (JsonNode col : columns) {
                String fk = col.path("fieldKey").asText(null);
                if ("Entity.Id".equalsIgnoreCase(fk)) {
                    return; // уже присутствует (регистронезависимо)
                }
            }
        }

        // 2. Добавляем {"fieldKey": "Entity.Id"} в массив columns
        if (config.isObject()) {
            ObjectNode obj = (ObjectNode) config;
            ArrayNode colsArray = obj.withArray("columns"); // создаст массив, если его нет
            ObjectNode newCol = colsArray.addObject();
            newCol.put("fieldKey", "Entity.Id");
        }
    }

    public V2LiveBookSchemaService(
            LiveBookService liveBookService,
            BusinessMetadataService businessMetadataService,

        this.liveBookService = liveBookService;
        this.businessMetadataService = businessMetadataService;
        this.tableNameResolver = tableNameResolver;

    public V2LiveBookSchema resolveSchema(JsonNode config) {
        LiveBookDto liveBook = resolveLiveBook(config);
        String tableName = tableNameResolver.resolve(config, liveBook.liveBookName());

        return buildSchema(liveBook, tableName);

    private LiveBookDto resolveLiveBook(JsonNode config) {
        List<LiveBookDto> liveBooks = liveBookService.findAll();

        if (config.hasNonNull("liveBookId")) {
            long liveBookId = config.get("liveBookId").asLong();

            return liveBooks.stream()
                    .filter(liveBook -> liveBook.liveBookId() != null)
                    .filter(liveBook -> liveBook.liveBookId() == liveBookId)
                    .findFirst()
                    .orElseThrow(() -> new IllegalArgumentException(
                            "LiveBook with id=" + liveBookId + " was not found"
                    ));
        }

        if (config.hasNonNull("liveBookName")) {
            String liveBookName = config.get("liveBookName").asText();

            return liveBooks.stream()
                    .filter(liveBook -> liveBookName.equals(liveBook.liveBookName()))
                    .findFirst()
                    .orElseThrow(() -> new IllegalArgumentException(
                            "LiveBook with name='" + liveBookName + "' was not found"
                    ));
        }

        throw new IllegalArgumentException(
                "v2/config requires liveBookId or liveBookName in data"
        );
    }

    private V2LiveBookSchema buildSchema(LiveBookDto liveBook, String tableName) {
        V2DataDictionaryPathResolver resolver =
                new V2DataDictionaryPathResolver(businessMetadataService.roots());

        Map<String, V2ResolvedLiveBookField> fieldsByPath = new LinkedHashMap<>();

        for (String fieldPath : liveBook.fields()) {
            try {
                V2ResolvedLiveBookField resolved = resolver.resolveScalar(fieldPath);
                fieldsByPath.put(fieldPath, resolved);
            } catch (Exception e) {
                throw new IllegalArgumentException(
                        "Cannot resolve LiveBook field '" + fieldPath +
                                "' for liveBook '" + liveBook.liveBookName() + "'",
                        e
                );
            }
        }

        return new V2LiveBookSchema(
                liveBook.liveBookId(),
                liveBook.liveBookName(),
                tableName,
                fieldsByPath
        );
    }

import com.fasterxml.jackson.core.type.TypeReference;
import com.fasterxml.jackson.databind.ObjectMapper;
import com.pivotService.livebook.dto.LiveBookDto;
import com.pivotService.livebook.exception.LiveBookAlreadyExistsException;
import com.pivotService.livebook.service.LiveBookService;
import org.springframework.beans.factory.annotation.Value;
import org.springframework.stereotype.Service;
import org.springframework.web.client.RestTemplate;

import java.util.*;
import java.util.concurrent.ConcurrentHashMap;
import java.util.function.Function;
import java.util.stream.Collectors;

@Service
public class RestLiveBookService implements LiveBookService {

    private final RestTemplate restTemplate;
    private final Map<Long, LiveBookDto> storage = new ConcurrentHashMap<>();
    private final ObjectMapper objectMapper;

    @Value("${app.getLiveBookURL}")
    private String getLiveBookURL;

    public RestLiveBookService(RestTemplate restTemplate, ObjectMapper objectMapper) {
        this.restTemplate = restTemplate;
        this.objectMapper = objectMapper;
    }

    @Override
    public LiveBookDto create(LiveBookDto request) {
        LiveBookDto liveBook = copy(request);

        LiveBookDto previous = storage.putIfAbsent(liveBook.liveBookId(), liveBook);

        if (previous != null) {
            throw new LiveBookAlreadyExistsException(
                    "LiveBook with id " + liveBook.liveBookId() + " already exists"
            );
        }

        return copy(liveBook);
    }

    private void updateAll () {
        String jsonResponse = restTemplate.getForObject(getLiveBookURL, String.class);
        List<LiveBookDto> liveBooks;
        try {
            // Парсим JSON в список
            liveBooks = objectMapper.readValue(
                    jsonResponse,
                    new TypeReference<List<LiveBookDto>>() {}
            );
        } catch (Exception e) {
            throw new RuntimeException(e);
        }
        storage.clear();
        storage.putAll(liveBooks.stream()
                .collect(Collectors.toMap(
                        LiveBookDto::liveBookId,
                        Function.identity()
                )));
    }

    @Override
    public List<LiveBookDto> findAll() {
        updateAll();
        return storage.values()
                .stream()
                .map(this::copy)
                .sorted(Comparator.comparing(LiveBookDto::liveBookId))
                .toList();
    }

    private LiveBookDto copy(LiveBookDto source) {
        return LiveBookDto.builder()
                .liveBookId(source.liveBookId())
                .liveBookName(source.liveBookName())
                .filter(source.filter())
                .fields(source.fields() == null ? List.of() : new ArrayList<>(source.fields()))
                .build();

package com.pivotService.resultBroker.response;

import com.pivotService.resultBroker.response.model.QueryData;
import org.springframework.stereotype.Component;

import java.sql.ResultSet;
import java.sql.ResultSetMetaData;
import java.sql.SQLException;
import java.util.ArrayList;
import java.util.LinkedHashMap;
import java.util.List;
import java.util.Map;

@Component
public class ResultSetExtractor {
    /**
     * Читает весь ResultSet в память и возвращает QueryData.
     * После вызова ResultSet полностью consumed, но НЕ закрыт.
     */
    public QueryData extract(ResultSet rs) throws SQLException {
        ResultSetMetaData metaData = rs.getMetaData();
        int columnCount = metaData.getColumnCount();

        List<String> columns = new ArrayList<>(columnCount);
        for (int i = 1; i <= columnCount; i++) {
            columns.add(metaData.getColumnLabel(i));
        }

        List<Map<String, Object>> rows = new ArrayList<>();
        while (rs.next()) {
            Map<String, Object> row = new LinkedHashMap<>(columnCount);
            for (int i = 1; i <= columnCount; i++) {
                row.put(columns.get(i - 1), rs.getObject(i));
            }
            rows.add(row);
        }

        return new QueryData(columns, rows);
    }
}

package com.pivotService.resultBroker.response.responseFactory;

import com.fasterxml.jackson.databind.JsonNode;
import com.pivotService.resultBroker.response.model.PivotColumnDefs;
import com.pivotService.resultBroker.response.model.QueryData;
import com.pivotService.resultBroker.response.model.QueryResponse;
import com.pivotService.sessionStorage.model.SessionRequest;
import lombok.extern.slf4j.Slf4j;
import org.springframework.core.annotation.Order;
import org.springframework.stereotype.Component;

import java.util.*;
import java.util.stream.Collectors;
import java.util.stream.StreamSupport;

@Slf4j
@Component
@Order(1)
public class PivotQueryResponseFactory extends AbstractQueryResponseFactory {

    @Override
    public boolean supports(SessionRequest sessionRequest) {
        JsonNode config = sessionRequest.config();
        if (config == null) return false;
        boolean pivotMode = config.has("pivotMode") && config.get("pivotMode").asBoolean();
        JsonNode pivotColumns = config.get("pivotColumns");
        return pivotMode && pivotColumns != null && pivotColumns.size() > 0;
    }

    @Override
    public QueryResponse create(QueryData data, SessionRequest sessionRequest) {
        JsonNode config = sessionRequest.config();

        List<String> rowGroupFieldKeys = extractFieldKeys(config.get("rowGroupColumns"));
        List<String> pivotFieldKeys = extractFieldKeys(config.get("pivotColumns"));
        List<ValueColumnDef> valueColumns = parseValueColumns(config.get("valueColumns"));
        JsonNode groupKeys = config.get("groupKeys");
        Map<String, JsonNode> groupKeysFieldValue = StreamSupport
                .stream(groupKeys.spliterator(), false)
                .collect(Collectors.toMap(
                        item -> item.get("fieldKey").asText(),
                        item -> item
                ));
        rowGroupFieldKeys.removeAll(groupKeysFieldValue.keySet());

        // Структуры для агрегации
        Map<String, Map<String, Object>> groupSamples = new LinkedHashMap<>(); // groupKey -> sample values
        Map<String, Map<String, Map<String, List<Object>>>> aggregationLists = new LinkedHashMap<>(); // groupKey -> pivotKey -> fieldKey -> values
        Set<String> pivotKeySet = new LinkedHashSet<>();

        for (Map<String, Object> row : data.rows()) {
            String groupKey = buildCompositeKey(rowGroupFieldKeys, row);
            String pivotKey = buildCompositeKey(pivotFieldKeys, row);

            groupSamples.putIfAbsent(groupKey, extractFieldValues(rowGroupFieldKeys, row));
            pivotKeySet.add(pivotKey);

            Map<String, Map<String, List<Object>>> pivotMap = aggregationLists.computeIfAbsent(groupKey, k -> new LinkedHashMap<>());
            Map<String, List<Object>> valueMap = pivotMap.computeIfAbsent(pivotKey, k -> new LinkedHashMap<>());
            for (ValueColumnDef vc : valueColumns) {
                Object value = row.get(vc.fieldKey);
                valueMap.computeIfAbsent(vc.fieldKey, k -> new ArrayList<>()).add(value);
            }
        }

        // Сортировка pivot ключей для стабильного порядка колонок
        List<String> sortedPivotKeys = new ArrayList<>(pivotKeySet);
        sortedPivotKeys.sort(Comparator.naturalOrder());

        // Превью отображаемых значений для каждого pivot (без fieldKey)
        Map<String, String> pivotDisplayValues = new LinkedHashMap<>();
        for (String pivotKey : sortedPivotKeys) {
            pivotDisplayValues.put(pivotKey, extractDisplayValueFromCompositeKey(pivotKey));
        }



        // columnDefs
        List<PivotColumnDefs> columnDefs = new ArrayList<>();
        for (String rgfk : rowGroupFieldKeys) {
            columnDefs.add(new PivotColumnDefs(rgfk));
        }
        for (String pivotKey : sortedPivotKeys) {
            String pivotDisplay = pivotDisplayValues.get(pivotKey);
            for (ValueColumnDef vc : valueColumns) {
                columnDefs.add(new PivotColumnDefs(pivotDisplay + " | " + capitalize(vc.fieldKey)) );
            }
        }
        columnDefs.add(new PivotColumnDefs("pivotId"));

        // rows
        List<Map<String, Object>> resultRows = new ArrayList<>();
        for (Map.Entry<String, Map<String, Object>> groupEntry : groupSamples.entrySet()) {
            String groupKey = groupEntry.getKey();
            Map<String, Object> groupValues = groupEntry.getValue();

            Map<String, Object> resultRow = new LinkedHashMap<>();
            resultRow.putAll(groupValues);
            resultRow.put("pivotId", buildPivotId(rowGroupFieldKeys, groupValues));

            Map<String, Map<String, List<Object>>> groupAgg = aggregationLists.get(groupKey);
            for (String pivotKey : sortedPivotKeys) {
                String pivotDisplay = pivotDisplayValues.get(pivotKey);
                Map<String, List<Object>> valueMap = groupAgg != null ? groupAgg.get(pivotKey) : null;

                for (ValueColumnDef vc : valueColumns) {
                    String columnKey = pivotDisplay + " | " + capitalize(vc.fieldKey);
                    Object aggregatedValue = null;
                    if (valueMap != null && valueMap.containsKey(vc.fieldKey)) {
                        aggregatedValue = computeAggregate(valueMap.get(vc.fieldKey), vc.aggFunc);
                    }
                    resultRow.put(columnKey, aggregatedValue);
                }
            }
            resultRows.add(resultRow);
        }

        return QueryResponse.builder()
                .columnDefs(columnDefs)
                .rows(resultRows)
                .success(true)
                .totalCount(resultRows.size())
                .requestId(getRequestId(sessionRequest))
                .build();
    }


    // ---------- вспомогательные методы ----------

    private List<ValueColumnDef> parseValueColumns(JsonNode valueColumnsNode) {
        List<ValueColumnDef> result = new ArrayList<>();
        if (valueColumnsNode != null && valueColumnsNode.isArray()) {
            for (JsonNode node : valueColumnsNode) {
                String fieldKey = node.get("fieldKey").asText();
                String aggFunc = node.has("aggFunc") ? node.get("aggFunc").asText() : "sum";
                result.add(new ValueColumnDef(fieldKey, aggFunc));
            }
        }
        return result;
    }

    private Map<String, Object> extractFieldValues(List<String> fieldKeys, Map<String, Object> row) {
        Map<String, Object> values = new LinkedHashMap<>();
        for (String key : fieldKeys) {
            values.put(key, row.get(key));
        }
        return values;
    }

    private String buildCompositeKey(List<String> fieldKeys, Map<String, Object> row) {
        return fieldKeys.stream()
                .map(k -> k + "=" + row.get(k))
                .collect(Collectors.joining("_"));
    }

    private String extractDisplayValueFromCompositeKey(String compositeKey) {
        if (compositeKey == null || compositeKey.isEmpty()) return "";
        String[] parts = compositeKey.split("_");
        List<String> values = new ArrayList<>();
        for (String part : parts) {
            String[] kv = part.split("=", 2);
            if (kv.length == 2) {
                values.add(kv[1]);
            }
        }
        return String.join(", ", values);
    }

    private String buildPivotId(List<String> rowGroupFieldKeys, Map<String, Object> row) {
        if (rowGroupFieldKeys.size() == 1) {
            String fieldKey = rowGroupFieldKeys.get(0);
            return fieldKey + "_" + row.get(fieldKey);
        }
        return rowGroupFieldKeys.stream()
                .map(k -> k + "_" + row.get(k))
                .collect(Collectors.joining("_"));
    }

    @SuppressWarnings({"unchecked", "rawtypes"})
    private Object computeAggregate(List<Object> values, String aggFunc) {
        if (values == null || values.isEmpty()) return null;

        switch (aggFunc.toLowerCase()) {
            case "count":
                return values.size();
            case "min":
                return values.stream()
                        .filter(Objects::nonNull)
                        .min((a, b) -> ((Comparable) a).compareTo(b))
                        .orElse(null);
            case "max":
                return values.stream()
                        .filter(Objects::nonNull)
                        .max((a, b) -> ((Comparable) a).compareTo(b))
                        .orElse(null);
            case "sum":
                double sum = values.stream()
                        .filter(Objects::nonNull)
                        .mapToDouble(v -> ((Number) v).doubleValue())
                        .sum();
                return sum;
            case "avg":
                double avg = values.stream()
                        .filter(Objects::nonNull)
                        .mapToDouble(v -> ((Number) v).doubleValue())
                        .average()
                        .orElse(0.0);
                return avg;
            default:
                return null;
        }
    }

    private String capitalize(String str) {
        if (str == null || str.isEmpty()) return str;
        return str.substring(0, 1).toUpperCase() + str.substring(1);
    }

    private String getRequestId(SessionRequest sessionRequest) {
        // Пытаемся взять из конфига или через метод requestId()
        try {
            // Если в SessionRequest есть метод requestId()
            return sessionRequest.requestId();
        } catch (Exception e) {
            // fallback: берём из config
            JsonNode config = sessionRequest.config();
            if (config != null && config.has("requestId")) {
                return config.get("requestId").asText();
            }
            return null;
        }
    }

    // ---------- внутренний класс ----------

    private static class ValueColumnDef {
        final String fieldKey;
        final String aggFunc;

        ValueColumnDef(String fieldKey, String aggFunc) {
            this.fieldKey = fieldKey;
            this.aggFunc = aggFunc;
        }
    }
}

package com.pivotService.resultBroker.response.responseFactory;

import com.fasterxml.jackson.databind.JsonNode;
import com.fasterxml.jackson.databind.node.ArrayNode;

import java.util.*;


public abstract class AbstractQueryResponseFactory implements QueryResponseFactory {

    /**
     * Извлекает список fieldKey из JSON-массива (rowGroupColumns / pivotColumns / valueColumns).
     */
    protected List<String> extractFieldKeys(JsonNode arrayNode) {
        List<String> keys = new ArrayList<>();
        if (arrayNode != null && arrayNode.isArray()) {
            for (JsonNode node : arrayNode) {
                keys.add(node.get("fieldKey").asText());
            }
        }
        return keys;
    }

    /**
     * В не зависимости от количества элемиентов для группировки всегда возвращаем строку вида: "groupId": "group-traderId-3"
     * или как в примере с двойной группировкой: "groupId": "group-traderId-3-legalEntityId-10"
     * в groupId попадают все значения из groupKeys и первое значение из rowGroupColumns
     */
    protected Object buildGroupId(List<String> groupFieldKeys, Map<String, Object> row, JsonNode groupKeys) {
        StringBuilder builder = new StringBuilder("group");

        // Безопасно получаем setGroupKeys — если groupKeys == null или не массив, используем пустой набор
        Set<String> setGroupKeys = (groupKeys != null && groupKeys.isArray())
                ? extractFieldKeySet((ArrayNode) groupKeys)
                : Collections.emptySet();

        // Добавляем элементы из groupKeys, если они есть
        if (groupKeys != null && groupKeys.isArray() && groupKeys.size() > 0) {
            for (JsonNode element : groupKeys) {
                String fieldKey = element.get("fieldKey").asText();
                String value = element.get("value").asText();
                builder.append("-").append(fieldKey).append("-").append(value);
            }
        }

        // Добавляем первое значение из rowGroupColumns, которого ещё нет в groupKeys
        for (String groupFieldKey : groupFieldKeys) {
            if (!setGroupKeys.contains(groupFieldKey) && row.containsKey(groupFieldKey)) {
                Object rawValue = row.get(groupFieldKey);
                String rowValue = (rawValue != null) ? rawValue.toString() : ""; // или "null"
                builder.append("-").append(groupFieldKey).append("-").append(rowValue);
                break;
            }
        }
        return builder.toString();
    }

    /**
     * Получает список уникальных значений из "fieldKey"
     */
    protected Set<String> extractFieldKeySet(ArrayNode arrayNode) {
        Set<String> result = new HashSet<>();
        for (JsonNode node : arrayNode) {
            JsonNode field = node.get("fieldKey");
            if (field != null) {
                result.add(field.asText());
            }
        }
        return result;
    }
}

@Slf4j
@Service
public class ExecuteSubscriptionQuery {
    private final SessionStorage sessionStorage;
    private final ClickHouseService clickHouseService;
    private final ResponseFactory responseFactory;
    private final ObjectMapper objectMapper;
    private final MessageSender responseSender;

    public ExecuteSubscriptionQuery(SessionStorage sessionStorage,
                                    ClickHouseService clickHouseService,
                                    ResponseFactory responseFactory,
                                    ObjectMapper objectMapper,
                                    MessageSender responseSender) {
        this.sessionStorage = sessionStorage;
        this.clickHouseService = clickHouseService;
        this.responseFactory = responseFactory;
        this.objectMapper = objectMapper;
        this.responseSender = responseSender;
    }

    @Scheduled(fixedRateString = "${app.clickhouse-scheduled-rate-millis}")
    private void executeSubscriptionQuery() {
        for (Map<Integer, Map<String, SessionRequest>> sessions : sessionStorage.getClientSubscription().values()) {
            for (Map<String, SessionRequest> viewGroupe : sessions.values()) {
                List<GroupedResponseItem> items = new ArrayList<>();
                String clientId = "";
                for (SessionRequest req : viewGroupe.values()) {
                    if (clientId.isEmpty()) {
                        clientId = req.clientId();
                    }


                    try {
                        QueryData queryData = clickHouseService.executeAndExtract(req.sql());
                        QueryResponse qr = responseFactory.buildResponse(queryData, req);
                        items.add(GroupedResponseItem.builder()
                                .requestId(req.requestId())
                                .rows(qr.getRows())
                                .totalCount(qr.getTotalCount())
                                .columnDefs(qr.getColumnDefs())
                                .build());
                    } catch (Exception e) {
                        log.error("Error executing SQL {} for group {}: {}", req.sql(), req.requestId(), e.getMessage());
                        items.add(GroupedResponseItem.builder()
                                .requestId(req.requestId())
                                .rows(Collections.emptyList())
                                .totalCount(0)
                                .build());
                    }
                }
                // Отправляем сгруппированный ответ клиенту
                GroupedResponse groupedResponse = GroupedResponse.builder()
                        .type("ReceiveData")
                        .success(true)
                        .data(items)
                        .build();
                try {
                    String json = objectMapper.writeValueAsString(groupedResponse);
                    responseSender.sendToSession(clientId, json);
                } catch (JsonProcessingException e) {
                    log.error("Error serializing grouped response for client {}", clientId, e);
                }
            }
        }

    }
}
