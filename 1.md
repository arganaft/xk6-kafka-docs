echo "🔧 Сборка RisingWave из исходников"

def ApplicationName       = "risingwave"
def RISINGWAVE_VERSION    = "v2.7.3"
def NexusGroupId          = 'ci01142241'
def NexusArtifactId       = "ci02631387/${ApplicationName}"
def dockerRegistryURL     = 'docker-dev.registry-ci.delta.sbrf.ru'

sh "mkdir -p docker_build/risingwave"
dir("docker_build/risingwave") {

    // ========== 1. Проверяем что базовые образы доступны ==========
    echo "🔍 Проверяем доступность базовых образов в корпоративном зеркале..."
    sh "docker pull sberosc.sigma.sbrf.ru/docker.io/docker.io/rust:1.87-alpine3.22 || echo 'ОШИБКА: образ rust не найден!'"
    sh "docker pull sberosc.sigma.sbrf.ru/docker.io/alpine:3.22.2 || echo 'ОШИБКА: образ alpine не найден!'"

    // ========== 2. Генерируем Dockerfile ==========
    echo "📝 Создаём Dockerfile..."
    writeFile file: 'Dockerfile', text: """\
########################################################
# Stage 1: Сборка из исходников на Alpine (musl libc)
########################################################
FROM sberosc.sigma.sbrf.ru/docker.io/docker.io/rust:1.87-alpine3.22 AS builder

# --- Корпоративное зеркало APK ---
RUN sed -i 's|https://dl-cdn.alpinelinux.org/alpine/|https://token:1234567890@sberosc.sigma.sbrf.ru/repo/apk/|g' /etc/apk/repositories

# --- Системные зависимости для сборки ---
RUN apk add --no-cache --no-check-certificate \\
    git \\
    ca-certificates \\
    openssl \\
    musl-dev \\
    gcc \\
    g++ \\
    cmake \\
    make \\
    protobuf-dev \\
    protoc \\
    pkgconfig \\
    openssl-dev \\
    openssl-libs-static \\
    perl \\
    python3 \\
    linux-headers \\
    librdkafka-dev \\
    zlib-dev \\
    lz4-dev \\
    zstd-dev \\
    snappy-dev \\
    curl

# --- Корпоративный SSL-сертификат ---
RUN openssl s_client -showcerts -connect sberosc.sigma.sbrf.ru:443 </dev/null 2>/dev/null \\
    | sed -ne '/-BEGIN CERTIFICATE-/,/-END CERTIFICATE-/p' \\
    > /usr/local/share/ca-certificates/sberosc.crt && update-ca-certificates

# --- Настройка Cargo: зеркало crates.io через корпоративный Nexus ---
RUN mkdir -p /root/.cargo && \\
    echo '[source.crates-io]'                                                                      > /root/.cargo/config.toml && \\
    echo 'replace-with = "corp-mirror"'                                                           >> /root/.cargo/config.toml && \\
    echo ''                                                                                       >> /root/.cargo/config.toml && \\
    echo '[source.corp-mirror]'                                                                   >> /root/.cargo/config.toml && \\
    echo 'registry = "sparse+https://token:1234567890@sberosc.sigma.sbrf.ru/repo/cargo/"'         >> /root/.cargo/config.toml && \\
    echo ''                                                                                       >> /root/.cargo/config.toml && \\
    echo '[net]'                                                                                  >> /root/.cargo/config.toml && \\
    echo 'git-fetch-with-cli = true'                                                              >> /root/.cargo/config.toml && \\
    echo ''                                                                                       >> /root/.cargo/config.toml && \\
    echo '[http]'                                                                                 >> /root/.cargo/config.toml && \\
    echo 'check-revoke = false'                                                                   >> /root/.cargo/config.toml && \\
    echo 'ssl-verify = false'                                                                     >> /root/.cargo/config.toml && \\
    echo ''                                                                                       >> /root/.cargo/config.toml && \\
    echo '[registries.corp-mirror]'                                                               >> /root/.cargo/config.toml && \\
    echo 'index = "sparse+https://token:1234567890@sberosc.sigma.sbrf.ru/repo/cargo/"'            >> /root/.cargo/config.toml && \\
    echo '--- Cargo config.toml ---' && cat /root/.cargo/config.toml

# --- Git: доступ к github.com через корпоративное зеркало ---
RUN git config --global http.sslVerify false && \\
    git config --global url."https://token:1234567890@sberosc.sigma.sbrf.ru/repo/github/".insteadOf "https://github.com/"

# --- Скачиваем исходники ---
WORKDIR /build
RUN echo '--- Клонируем RisingWave ${RISINGWAVE_VERSION} ---' && \\
    git clone --depth 1 --branch ${RISINGWAVE_VERSION} \\
    https://github.com/risingwavelabs/risingwave.git . && \\
    echo '--- Клонирование завершено ---' && ls -la

# --- Переменные окружения для сборки ---
ENV OPENSSL_STATIC=1
ENV OPENSSL_LIB_DIR=/usr/lib
ENV OPENSSL_INCLUDE_DIR=/usr/include
ENV PROTOC=/usr/bin/protoc
ENV RUSTFLAGS="-C target-feature=-crt-static"

# --- Сборка RisingWave ---
RUN echo '--- Начало компиляции ---' && \\
    cargo build --release --bin risingwave && \\
    echo '--- Компиляция завершена ---'

RUN ls -lh /build/target/release/risingwave && \\
    file /build/target/release/risingwave

########################################################
# Stage 2: Минимальный рантайм на Alpine
########################################################
FROM sberosc.sigma.sbrf.ru/docker.io/alpine:3.22.2

RUN sed -i 's|https://dl-cdn.alpinelinux.org/alpine/|https://token:1234567890@sberosc.sigma.sbrf.ru/repo/apk/|g' /etc/apk/repositories

RUN apk add --no-cache --no-check-certificate \\
    ca-certificates \\
    libgcc \\
    libstdc++ \\
    openssl \\
    zlib \\
    lz4-libs \\
    zstd-libs \\
    snappy \\
    librdkafka

COPY --from=builder /build/target/release/risingwave /usr/bin/risingwave

RUN adduser -D -s /sbin/nologin rwuser
RUN mkdir -p /data /risingwave/state_store && \\
    chown -R rwuser:rwuser /data /risingwave
RUN rm -rf /usr/share/doc /usr/share/man /var/cache/apk/*

USER rwuser
WORKDIR /home/rwuser

EXPOSE 4566 5691

ENTRYPOINT ["risingwave"]
CMD ["single_node"]
""".stripIndent()

    // ========== 3. Проверяем Dockerfile ==========
    echo "📄 Содержимое Dockerfile:"
    sh "cat -n Dockerfile"
    sh "echo 'Строк в Dockerfile:' && wc -l Dockerfile"

    // ========== 4. Чистим предыдущие кэши ==========
    sh "docker rmi ${dockerRegistryURL}/${NexusGroupId}/${NexusArtifactId}:${RISINGWAVE_VERSION} 2>/dev/null || true"

    // ========== 5. Собираем ==========
    echo "⏳ Запускаем сборку (BuildKit ОТКЛЮЧЁН, кэш ОТКЛЮЧЁН)..."
    sh """
        DOCKER_BUILDKIT=0 docker build \\
            --pull \\
            --no-cache \\
            --progress=plain \\
            -t ${dockerRegistryURL}/${NexusGroupId}/${NexusArtifactId}:${RISINGWAVE_VERSION} \\
            . 2>&1 | tee build.log
    """

    // ========== 6. Проверяем результат ==========
    echo "📏 Размер образа:"
    sh "docker images | grep risingwave || echo 'ОБРАЗ НЕ НАЙДЕН!'"

    echo "🔍 Проверяем бинарник:"
    sh "docker run --rm ${dockerRegistryURL}/${NexusGroupId}/${NexusArtifactId}:${RISINGWAVE_VERSION} --version"

    // ========== 7. Пушим ==========
    sh "docker image push ${dockerRegistryURL}/${NexusGroupId}/${NexusArtifactId}:${RISINGWAVE_VERSION}"

    echo "✅ Готово"
}
