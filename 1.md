echo "🔧 Сборка RisingWave из исходников"

def ApplicationName       = "risingwave"
def RISINGWAVE_VERSION    = "v2.7.3"
def NexusGroupId          = 'ci01142241'
def NexusArtifactId       = "ci02631387/${ApplicationName}"
def dockerRegistryURL     = 'docker-dev.registry-ci.delta.sbrf.ru'

sh "mkdir -p docker_build/risingwave"
dir("docker_build/risingwave") {

    // ========== 1. Проверяем какие образы rust доступны ==========
    echo "=" * 60
    echo "📥 ШАГ 1: Ищем доступные образы rust:*-alpine в зеркале"
    echo "=" * 60

    // Попробуем несколько версий — чем новее, тем лучше
    sh """
        echo '--- Пробуем rust:1.85-alpine ---'
        docker pull sberosc.sigma.sbrf.ru/docker.io/rust:1.85-alpine 2>&1 && echo '✅ 1.85 доступен' || echo '❌ 1.85 недоступен'

        echo '--- Пробуем rust:1.83-alpine ---'
        docker pull sberosc.sigma.sbrf.ru/docker.io/rust:1.83-alpine 2>&1 && echo '✅ 1.83 доступен' || echo '❌ 1.83 недоступен'

        echo '--- Пробуем rust:1.81-alpine ---'
        docker pull sberosc.sigma.sbrf.ru/docker.io/rust:1.81-alpine 2>&1 && echo '✅ 1.81 доступен' || echo '❌ 1.81 недоступен'

        echo '--- Пробуем rust:1.79-alpine (уже скачан) ---'
        docker pull sberosc.sigma.sbrf.ru/docker.io/rust:1.79-alpine 2>&1 && echo '✅ 1.79 доступен' || echo '❌ 1.79 недоступен'

        echo '--- Пробуем rust:1.87-alpine ---'
        docker pull sberosc.sigma.sbrf.ru/docker.io/rust:1.87-alpine 2>&1 && echo '✅ 1.87 доступен' || echo '❌ 1.87 недоступен'

        echo '--- Пробуем rust:1.86-alpine ---'
        docker pull sberosc.sigma.sbrf.ru/docker.io/rust:1.86-alpine 2>&1 && echo '✅ 1.86 доступен' || echo '❌ 1.86 недоступен'

        echo '--- Пробуем rust:1.84-alpine ---'
        docker pull sberosc.sigma.sbrf.ru/docker.io/rust:1.84-alpine 2>&1 && echo '✅ 1.84 доступен' || echo '❌ 1.84 недоступен'

        echo '--- Пробуем rust:1.82-alpine ---'
        docker pull sberosc.sigma.sbrf.ru/docker.io/rust:1.82-alpine 2>&1 && echo '✅ 1.82 доступен' || echo '❌ 1.82 недоступен'

        echo '--- Пробуем rust:1.80-alpine ---'
        docker pull sberosc.sigma.sbrf.ru/docker.io/rust:1.80-alpine 2>&1 && echo '✅ 1.80 доступен' || echo '❌ 1.80 недоступен'
    """

    // ========== 2. Проверяем Alpine в rust образе ==========
    echo "=" * 60
    echo "🔍 ШАГ 2: Смотрим что внутри rust:1.79-alpine"
    echo "=" * 60

    sh """
        echo '--- Версия Rust ---'
        docker run --rm sberosc.sigma.sbrf.ru/docker.io/rust:1.79-alpine rustc --version

        echo '--- Версия Alpine внутри ---'
        docker run --rm sberosc.sigma.sbrf.ru/docker.io/rust:1.79-alpine cat /etc/alpine-release

        echo '--- Текущие APK repos ---'
        docker run --rm sberosc.sigma.sbrf.ru/docker.io/rust:1.79-alpine cat /etc/apk/repositories

        echo '--- libxml2 уже установлен? ---'
        docker run --rm sberosc.sigma.sbrf.ru/docker.io/rust:1.79-alpine apk list --installed 2>/dev/null | grep -i libxml || echo 'не найден через apk list'
        docker run --rm sberosc.sigma.sbrf.ru/docker.io/rust:1.79-alpine find / -name 'libxml2*' 2>/dev/null | head -10 || echo 'файлы не найдены'

        echo '--- LLVM уже установлен? ---'
        docker run --rm sberosc.sigma.sbrf.ru/docker.io/rust:1.79-alpine apk list --installed 2>/dev/null | grep -i llvm || echo 'не найден'

        echo '--- Все предустановленные пакеты ---'
        docker run --rm sberosc.sigma.sbrf.ru/docker.io/rust:1.79-alpine apk list --installed 2>/dev/null | wc -l
    """

    // ========== 3. Генерируем Dockerfile ==========
    echo "=" * 60
    echo "📝 ШАГ 3: Создаём Dockerfile"
    echo "=" * 60

    // ВАЖНО: замените 1.79 на максимальную доступную версию из ШАГ 1
    writeFile file: 'Dockerfile', text: """\
########################################################
# Stage 1: Сборка на rust:alpine
# Rust + LLVM + libxml2 = УЖЕ УСТАНОВЛЕНЫ в базовом образе
# Не нужен apk add rust → не нужен libxml2 из зеркала!
########################################################
FROM sberosc.sigma.sbrf.ru/docker.io/rust:1.79-alpine AS builder

RUN echo '========== STAGE 1: Начало ==========' && \\
    echo '--- Rust (предустановлен) ---' && rustc --version && cargo --version && \\
    echo '--- Alpine версия ---' && cat /etc/alpine-release

# --- КЛЮЧЕВОЙ МОМЕНТ: ---
# Alpine 3.20 repos (из rust:1.79-alpine) ЗАБЛОКИРОВАНЫ в корпоративном зеркале
# Переключаемся НАПРЯМУЮ на v3.23 (который работает) для остальных пакетов
RUN echo '--- Старые repos ---' && cat /etc/apk/repositories && \\
    echo 'https://token:1234567890@sberosc.sigma.sbrf.ru/repo/apk/v3.23/main' > /etc/apk/repositories && \\
    echo 'https://token:1234567890@sberosc.sigma.sbrf.ru/repo/apk/v3.23/community' >> /etc/apk/repositories && \\
    echo '--- Новые repos ---' && cat /etc/apk/repositories && \\
    echo '✅ APK repos переключены на v3.23'

# --- Доустанавливаем ТОЛЬКО build-зависимости ---
# rust и cargo НЕ указываем — они уже есть в базовом образе!
RUN echo '⏳ Устанавливаем build-зависимости...' && \\
    apk add --no-cache --no-check-certificate \\
        git \\
        ca-certificates \\
        openssl \\
        musl-dev \\
        gcc \\
        g++ \\
        cmake \\
        make \\
        protobuf-dev \\
        pkgconfig \\
        openssl-dev \\
        openssl-libs-static \\
        perl \\
        linux-headers \\
        librdkafka-dev \\
        zlib-dev \\
        lz4-dev \\
        zstd-dev \\
        snappy-dev \\
        curl \\
        file && \\
    echo '✅ Build-зависимости установлены' && \\
    echo '--- protoc ---' && protoc --version

# --- Корпоративный SSL-сертификат ---
RUN openssl s_client -showcerts -connect sberosc.sigma.sbrf.ru:443 </dev/null 2>/dev/null \\
    | sed -ne '/-BEGIN CERTIFICATE-/,/-END CERTIFICATE-/p' \\
    > /usr/local/share/ca-certificates/sberosc.crt && update-ca-certificates && \\
    echo '✅ SSL-сертификат установлен'

# --- Настройка Cargo: зеркало crates.io ---
RUN mkdir -p /root/.cargo && \\
    echo '[source.crates-io]'                                                                      > /root/.cargo/config.toml && \\
    echo 'replace-with = "corp-mirror"'                                                           >> /root/.cargo/config.toml && \\
    echo ''                                                                                       >> /root/.cargo/config.toml && \\
    echo '[source.corp-mirror]'                                                                   >> /root/.cargo/config.toml && \\
    echo 'registry = "sparse+https://token:1234567890@sberosc.sigma.sbrf.ru/repo/cargo/"'         >> /root/.cargo/config.toml && \\
    echo ''                                                                                       >> /root/.cargo/config.toml && \\
    echo '[net]'                                                                                  >> /root/.cargo/config.toml && \\
    echo 'git-fetch-with-cli = true'                                                              >> /root/.cargo/config.toml && \\
    echo ''                                                                                       >> /root/.cargo/config.toml && \\
    echo '[http]'                                                                                 >> /root/.cargo/config.toml && \\
    echo 'check-revoke = false'                                                                   >> /root/.cargo/config.toml && \\
    echo 'ssl-verify = false'                                                                     >> /root/.cargo/config.toml && \\
    echo ''                                                                                       >> /root/.cargo/config.toml && \\
    echo '[registries.corp-mirror]'                                                               >> /root/.cargo/config.toml && \\
    echo 'index = "sparse+https://token:1234567890@sberosc.sigma.sbrf.ru/repo/cargo/"'            >> /root/.cargo/config.toml && \\
    echo '✅ Cargo config.toml:' && cat /root/.cargo/config.toml

# --- Git: github.com через корпоративное зеркало ---
RUN git config --global http.sslVerify false && \\
    git config --global url."https://token:1234567890@sberosc.sigma.sbrf.ru/repo/github/".insteadOf "https://github.com/" && \\
    echo '✅ Git настроен для корпоративного зеркала'

# --- Скачиваем исходники ---
WORKDIR /build
RUN echo '⏳ Клонируем RisingWave ${RISINGWAVE_VERSION}...' && \\
    git clone --depth 1 --branch ${RISINGWAVE_VERSION} \\
        https://github.com/risingwavelabs/risingwave.git . && \\
    echo '✅ Клонирование завершено' && ls -la && \\
    echo '--- rust-toolchain.toml (требуемая версия Rust) ---' && \\
    cat rust-toolchain.toml 2>/dev/null || cat rust-toolchain 2>/dev/null || echo 'Файл не найден'

# --- Переменные для сборки ---
ENV OPENSSL_STATIC=1
ENV OPENSSL_LIB_DIR=/usr/lib
ENV OPENSSL_INCLUDE_DIR=/usr/include
ENV PROTOC=/usr/bin/protoc
ENV RUSTFLAGS="-C target-feature=-crt-static"

# --- Сборка RisingWave ---
RUN echo '========================================' && \\
    echo '⏳ НАЧАЛО КОМПИЛЯЦИИ' && \\
    echo '========================================' && \\
    echo "Rust: \$(rustc --version)" && \\
    echo "Cargo: \$(cargo --version)" && \\
    echo "Время начала: \$(date)" && \\
    cargo build --release --bin risingwave && \\
    echo "Время окончания: \$(date)" && \\
    echo '========================================' && \\
    echo '✅ КОМПИЛЯЦИЯ ЗАВЕРШЕНА' && \\
    echo '========================================'

RUN echo '--- Бинарник ---' && \\
    ls -lh /build/target/release/risingwave && \\
    file /build/target/release/risingwave && \\
    echo '✅ Бинарник собран'

########################################################
# Stage 2: Минимальный рантайм на Alpine
########################################################
FROM sberosc.sigma.sbrf.ru/docker.io/alpine:3.23.3

RUN echo '========== STAGE 2: Рантайм =========='

RUN echo 'https://token:1234567890@sberosc.sigma.sbrf.ru/repo/apk/v3.23/main' > /etc/apk/repositories && \\
    echo 'https://token:1234567890@sberosc.sigma.sbrf.ru/repo/apk/v3.23/community' >> /etc/apk/repositories && \\
    echo '✅ APK repos настроены'

RUN echo '⏳ Устанавливаем runtime-пакеты...' && \\
    apk add --no-cache --no-check-certificate \\
        ca-certificates \\
        libgcc \\
        libstdc++ \\
        openssl \\
        zlib \\
        lz4-libs \\
        zstd-libs \\
        snappy \\
        openjdk17-jre-headless && \\
    echo '✅ Runtime-пакеты установлены' && \\
    java -version 2>&1

COPY --from=builder /build/target/release/risingwave /usr/bin/risingwave
RUN echo '⏳ Проверяем бинарник...' && \\
    ls -lh /usr/bin/risingwave && \\
    file /usr/bin/risingwave && \\
    echo '✅ Бинарник скопирован'

ENV JAVA_HOME=/usr/lib/jvm/java-17-openjdk
ENV PATH="\${JAVA_HOME}/bin:\${PATH}"

RUN adduser -D -s /sbin/nologin rwuser && \\
    mkdir -p /data /risingwave/state_store && \\
    chown -R rwuser:rwuser /data /risingwave && \\
    rm -rf /usr/share/doc /usr/share/man /var/cache/apk/* && \\
    echo '✅ Пользователь и каталоги созданы'

RUN echo '========== ФИНАЛЬНАЯ ПРОВЕРКА ==========' && \\
    echo '--- ОС ---' && cat /etc/os-release | head -3 && \\
    echo '--- Java ---' && java -version 2>&1 && \\
    echo '--- Пакеты ---' && apk list --installed 2>/dev/null | wc -l && \\
    echo '--- Размер бинарника ---' && ls -lh /usr/bin/risingwave && \\
    echo '========== ✅ ОБРАЗ СОБРАН =========='

USER rwuser
WORKDIR /home/rwuser
EXPOSE 4566 5691 5690 50051
ENTRYPOINT ["risingwave"]
CMD ["single_node"]
""".stripIndent()

    sh "cat -n Dockerfile"
    sh "wc -l Dockerfile"

    // ========== 4. Сборка ==========
    echo "=" * 60
    echo "⏳ ШАГ 4: Запускаем сборку"
    echo "=" * 60

    sh "docker rmi ${dockerRegistryURL}/${NexusGroupId}/${NexusArtifactId}:${RISINGWAVE_VERSION} 2>/dev/null || true"

    def startTime = System.currentTimeMillis()

    sh """
        DOCKER_BUILDKIT=0 docker build \\
            --pull \\
            --no-cache \\
            --progress=plain \\
            -t ${dockerRegistryURL}/${NexusGroupId}/${NexusArtifactId}:${RISINGWAVE_VERSION} \\
            . 2>&1 | tee build.log
    """

    def duration = (System.currentTimeMillis() - startTime) / 1000
    echo "⏱️ Сборка заняла: ${duration} секунд (${duration/60} минут)"

    // ========== 5. Проверка ==========
    echo "=" * 60
    echo "🔍 ШАГ 5: Проверяем результат"
    echo "=" * 60

    sh "docker images | grep risingwave || echo '❌ ОБРАЗ НЕ НАЙДЕН'"
    sh """
        docker run --rm ${dockerRegistryURL}/${NexusGroupId}/${NexusArtifactId}:${RISINGWAVE_VERSION} --version && \\
            echo '✅ risingwave работает' || echo '❌ risingwave НЕ работает'
    """
    sh """
        docker run --rm --entrypoint sh ${dockerRegistryURL}/${NexusGroupId}/${NexusArtifactId}:${RISINGWAVE_VERSION} \\
            -c 'java -version 2>&1' && echo '✅ Java работает' || echo '❌ Java НЕ работает'
    """

    // ========== 6. Пуш ==========
    echo "=" * 60
    echo "📤 ШАГ 6: Пушим образ"
    echo "=" * 60

    sh """
        docker image push ${dockerRegistryURL}/${NexusGroupId}/${NexusArtifactId}:${RISINGWAVE_VERSION} && \\
            echo '✅ Образ запушен' || echo '❌ Ошибка пуша'
    """

    echo "✅ ГОТОВО: ${ApplicationName}:${RISINGWAVE_VERSION}"
}
