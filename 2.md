// ============================================================
// 1. Клонируем исходники RisingWave
// ============================================================
sh "mkdir -p docker_build/risingwave"
dir("docker_build/risingwave") {

    // Клонируем через внутренний зеркало GitHub или напрямую
    sh """
        git clone --depth 1 --branch ${RISINGWAVE_VERSION} \
            https://github.com/risingwavelabs/risingwave.git source
    """

    // ============================================================
    // 2. Создаём оптимизированный Dockerfile
    // ============================================================
    writeFile file: 'Dockerfile.minimal', text: '''
# ================================================================
# STAGE 1: Базовый образ для сборки (build-only, не попадёт в прод)
# ================================================================
FROM sberosc.sigma.sbrf.ru/docker.io/ubuntu:24.04 AS build-base

ENV LANG=en_US.utf8
ENV DEBIAN_FRONTEND=noninteractive

# >>> Настройка внутреннего APT-зеркала <<<
# Замените URL на ваш внутренний Ubuntu-репозиторий
RUN sed -i "s|http://archive.ubuntu.com/ubuntu|https://sberosc.sigma.sbrf.ru/repo/ubuntu|g" /etc/apt/sources.list.d/ubuntu.sources 2>/dev/null || true && \\
    sed -i "s|http://security.ubuntu.com/ubuntu|https://sberosc.sigma.sbrf.ru/repo/ubuntu|g" /etc/apt/sources.list.d/ubuntu.sources 2>/dev/null || true

RUN apt-get update && apt-get -y install --no-install-recommends \\
    ca-certificates build-essential libsasl2-dev openssl pkg-config \\
    curl git libblas-dev liblapack-dev libomp-dev \\
    make cmake protobuf-compiler bash lld unzip rsync \\
    && rm -rf /var/lib/apt/lists/*


# ================================================================
# STAGE 2: Rust toolchain + сборка бинарника
# ================================================================
FROM build-base AS rust-builder

# --- Установка Node.js (для dashboard) ---
ENV NVM_DIR=/root/.nvm
ENV NODE_VERSION=20.11.1
RUN curl -o- https://raw.githubusercontent.com/nvm-sh/nvm/v0.39.7/install.sh | bash && \\
    . $NVM_DIR/nvm.sh && \\
    nvm install $NODE_VERSION
ENV PATH=$NVM_DIR/versions/node/v$NODE_VERSION/bin:$PATH

# >>> Настройка внутреннего NPM-реестра <<<
RUN npm config set registry https://sberosc.sigma.sbrf.ru/repo/npm/ && \\
    npm config set strict-ssl false

SHELL ["/bin/bash", "-c"]

# --- Установка Rust ---
RUN curl --proto "=https" --tlsv1.2 -sSf https://sh.rustup.rs | \\
    sh -s -- --no-modify-path --default-toolchain none -y
ENV PATH=/root/.cargo/bin:$PATH
ENV CARGO_INCREMENTAL=0

# >>> Настройка внутреннего Cargo/Crates-реестра <<<
# Вариант A: Через CARGO_REGISTRIES (если есть sparse registry proxy)
# ENV CARGO_REGISTRIES_CRATES_IO_INDEX="sparse+https://sberosc.sigma.sbrf.ru/repo/crates/"
#
# Вариант B: Через .cargo/config.toml
RUN mkdir -p /root/.cargo && \\
    cat >> /root/.cargo/config.toml <<'CARGO_EOF'
[source.crates-io]
replace-with = "internal-mirror"

[source.internal-mirror]
# Замените URL на ваш внутренний sparse-реестр crates.io
registry = "sparse+https://sberosc.sigma.sbrf.ru/repo/crates/"

[http]
check-revoke = false

[net]
git-fetch-with-cli = true
CARGO_EOF

# Добавляем внутренние CA-сертификаты
RUN openssl s_client -showcerts -connect sberosc.sigma.sbrf.ru:443 </dev/null 2>/dev/null | \\
    sed -ne '/-BEGIN CERTIFICATE-/,/-END CERTIFICATE-/p' > /usr/local/share/ca-certificates/sberosc.crt && \\
    update-ca-certificates

COPY source/rust-toolchain /risingwave/rust-toolchain
WORKDIR /risingwave
RUN rustup show && \\
    rustup component add rustfmt && \\
    rustup target add wasm32-wasip1

# --- Копируем исходники и собираем ---
COPY source/ /risingwave/

ENV ENABLE_BUILD_DASHBOARD=1
ENV OPENSSL_STATIC=1

ARG GIT_SHA=""
ENV GIT_SHA=$GIT_SHA
ARG CARGO_PROFILE=production
ENV CARGO_PROFILE=$CARGO_PROFILE

RUN cargo fetch && \\
    cargo build -p risingwave_cmd_all \\
        --profile ${CARGO_PROFILE} \\
        --features "rw-static-link" \\
        --features udf && \\
    mkdir -p /output/bin && \\
    mv /risingwave/target/${CARGO_PROFILE}/risingwave /output/bin/ && \\
    # Опционально: скопировать debug symbols отдельно (НЕ в продакшн-образ)
    (mv /risingwave/target/${CARGO_PROFILE}/risingwave.dwp /output/bin/ 2>/dev/null || true) && \\
    cargo clean


# ================================================================
# STAGE 3: Java Connector (ОПЦИОНАЛЬНО — раскомментируйте если нужен CDC/JDBC)
# ================================================================
# Если вам НЕ нужны Java-коннекторы (JDBC source/sink, CDC),
# оставьте этот блок закомментированным — это уберёт JRE из финального образа
# и радикально снизит количество CVE.

# FROM build-base AS java-builder
#
# RUN apt-get update && apt-get -y install --no-install-recommends \\
#     openjdk-17-jdk maven \\
#     && rm -rf /var/lib/apt/lists/*
#
# COPY source/java /risingwave/java/
# COPY source/Cargo.toml /risingwave/Cargo.toml
# COPY source/Cargo.lock /risingwave/Cargo.lock
# COPY source/src /risingwave/src
# WORKDIR /risingwave/java
#
# # >>> Настройка внутреннего Maven-зеркала <<<
# RUN mkdir -p /root/.m2 && \\
#     cat > /root/.m2/settings.xml <<'MVN_EOF'
# <settings>
#   <mirrors>
#     <mirror>
#       <id>internal-mirror</id>
#       <mirrorOf>*</mirrorOf>
#       <url>https://sberosc.sigma.sbrf.ru/repo/maven/</url>
#     </mirror>
#   </mirrors>
# </settings>
# MVN_EOF
#
# RUN mvn -B package -Dmaven.test.skip=true -Dno-build-rust && \\
#     mkdir -p /output/connector-node && \\
#     tar -zxvf /risingwave/java/connector-node/assembly/target/risingwave-connector-1.0.0.tar.gz \\
#         -C /output/connector-node


# ================================================================
# STAGE 4: ФИНАЛЬНЫЙ МИНИМАЛЬНЫЙ ПРОДАКШН-ОБРАЗ
# ================================================================
FROM sberosc.sigma.sbrf.ru/docker.io/alpine:3.22.2 AS production

# >>> Настройка внутреннего APK-репозитория <<<
RUN sed -i "s|https://dl-cdn.alpinelinux.org/alpine/|https://token:1234567890@sberosc.sigma.sbrf.ru/repo/apk/|g" /etc/apk/repositories

# Устанавливаем ТОЛЬКО необходимые runtime-пакеты
RUN apk add --no-cache --no-check-certificate \\
    ca-certificates \\
    gcompat \\
    libstdc++ \\
    libgcc \\
    && rm -rf /var/cache/apk/* /tmp/*

# ----------------------------------------------------------
# Раскомментируйте ТОЛЬКО если нужен Java Connector:
# RUN apk add --no-cache --no-check-certificate openjdk17-jre-headless
# ----------------------------------------------------------

# Копируем ТОЛЬКО бинарник RisingWave (без debug-символов!)
COPY --from=rust-builder /output/bin/risingwave /risingwave/bin/risingwave
RUN chmod +x /risingwave/bin/risingwave

# ----------------------------------------------------------
# Раскомментируйте ТОЛЬКО если нужен Java Connector:
# RUN mkdir -p /risingwave/bin/connector-node
# COPY --from=java-builder /output/connector-node /risingwave/bin/connector-node
# ENV CONNECTOR_LIBS_PATH=/risingwave/bin/connector-node/libs
# ----------------------------------------------------------

# === Безопасность: non-root пользователь ===
RUN addgroup -S risingwave && adduser -S -G risingwave -s /bin/sh risingwave && \\
    chown -R risingwave:risingwave /risingwave
USER risingwave

# === Минимальные переменные окружения ===
ENV PLAYGROUND_PROFILE=docker-playground
ENV IN_CONTAINER=1

# === Healthcheck ===
HEALTHCHECK --interval=30s --timeout=5s --start-period=10s --retries=3 \\
    CMD ["/risingwave/bin/risingwave", "--version"]

CMD ["single_node"]
'''

    sh "cat Dockerfile.minimal"

    // ============================================================
    // 3. Собираем Docker-образ
    // ============================================================
    echo "🔨 Собираем оптимизированный образ RisingWave"
    sh """
        docker build \\
            -f Dockerfile.minimal \\
            --build-arg GIT_SHA=\$(cd source && git rev-parse HEAD) \\
            --build-arg CARGO_PROFILE=production \\
            --no-cache \\
            -t ${dockerRegistryURL}/${NexusGroupId}/${NexusArtifactId}:${IMAGE_TAG} \\
            -t ${dockerRegistryURL}/${NexusGroupId}/${NexusArtifactId}:latest \\
            .
    """

    // ============================================================
    // 4. Сканирование на уязвимости (опционально)
    // ============================================================
    echo "🔍 Сканируем образ на уязвимости"
    sh """
        docker run --rm \\
            -v /var/run/docker.sock:/var/run/docker.sock \\
            sberosc.sigma.sbrf.ru/docker.io/aquasec/trivy:latest \\
            image --severity HIGH,CRITICAL --no-progress \\
            ${dockerRegistryURL}/${NexusGroupId}/${NexusArtifactId}:${IMAGE_TAG} || true
    """

    // ============================================================
    // 5. Пушим в реестр
    // ============================================================
    echo "📦 Пушим образ в Nexus"
    sh "docker image push ${dockerRegistryURL}/${NexusGroupId}/${NexusArtifactId}:${IMAGE_TAG}"
    sh "docker image push ${dockerRegistryURL}/${NexusGroupId}/${NexusArtifactId}:latest"

    // ============================================================
    // 6. Очистка
    // ============================================================
    echo "🧹 Очистка"
    sh """
        docker rmi ${dockerRegistryURL}/${NexusGroupId}/${NexusArtifactId}:${IMAGE_TAG} || true
        docker rmi ${dockerRegistryURL}/${NexusGroupId}/${NexusArtifactId}:latest || true
        docker builder prune -f || true
    """
}
