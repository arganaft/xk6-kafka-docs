echo "🔧 Сборка RisingWave из исходников (Rust + Alpine)"

def ApplicationName       = "risingwave"
def RISINGWAVE_VERSION    = "v2.7.3"
def NexusGroupId          = 'ci01142241'
def NexusArtifactId       = "ci02631387/${ApplicationName}"
def dockerRegistryURL     = 'docker-dev.registry-ci.delta.sbrf.ru'

sh "mkdir -p docker_build/risingwave"
dir("docker_build/risingwave") {

    echo "📝 Создаём Dockerfile (multi-stage build from source)"
    writeFile file: 'Dockerfile', text: """
    ########################################################
    # Stage 1: Сборка из исходников на Alpine (musl libc)
    # Аналогично k6: golang:alpine → rust:alpine
    ########################################################
    FROM sberosc.sigma.sbrf.ru/docker.io/docker.io/rust:1.87-alpine3.22 AS builder

    # --- Корпоративное зеркало APK (как в вашем k6 скрипте) ---
    RUN sed -i 's|https://dl-cdn.alpinelinux.org/alpine/|https://token:1234567890@sberosc.sigma.sbrf.ru/repo/apk/|g' /etc/apk/repositories

    # --- Системные зависимости для сборки ---
    RUN apk add --no-cache --no-check-certificate \\
        git \\
        ca-certificates \\
        openssl \\
        musl-dev \\
        gcc \\
        g++ \\
        cmake \\
        make \\
        protobuf-dev \\
        protoc \\
        pkgconfig \\
        openssl-dev \\
        openssl-libs-static \\
        perl \\
        python3 \\
        linux-headers \\
        librdkafka-dev \\
        zlib-dev \\
        lz4-dev \\
        zstd-dev \\
        snappy-dev \\
        curl

    # --- Корпоративный SSL-сертификат (как в вашем k6 скрипте) ---
    RUN openssl s_client -showcerts -connect sberosc.sigma.sbrf.ru:443 </dev/null 2>/dev/null \\
        | sed -ne '/-BEGIN CERTIFICATE-/,/-END CERTIFICATE-/p' \\
        > /usr/local/share/ca-certificates/sberosc.crt && update-ca-certificates

    # --- Настройка Cargo: зеркало crates.io через корпоративный Nexus ---
    # (аналог GOPROXY для Go — ваш Nexus проксирует crates.io)
    RUN mkdir -p /root/.cargo && cat > /root/.cargo/config.toml << 'CARGO_EOF'
    [source.crates-io]
    replace-with = "corp-mirror"

    [source.corp-mirror]
    registry = "sparse+https://token:1234567890@sberosc.sigma.sbrf.ru/repo/cargo/"

    [net]
    git-fetch-with-cli = true

    [http]
    check-revoke = false
    ssl-verify = false

    [registries.corp-mirror]
    index = "sparse+https://token:1234567890@sberosc.sigma.sbrf.ru/repo/cargo/"
    CARGO_EOF

    # --- Настройка Git для доступа к github.com через корпоративное зеркало ---
    # Cargo.toml может содержать git-зависимости вида github.com/...
    # Подставляем корпоративное зеркало GitHub (аналог GOPROXY для git-зависимостей)
    RUN git config --global http.sslVerify false && \\
        git config --global url."https://token:1234567890@sberosc.sigma.sbrf.ru/repo/github/".insteadOf "https://github.com/"

    # --- Скачиваем исходники ---
    # git clone работает через корпоративное зеркало GitHub (см. git config выше)
    WORKDIR /build
    RUN git clone --depth 1 --branch ${RISINGWAVE_VERSION} \\
        https://github.com/risingwavelabs/risingwave.git .

    # --- Переменные окружения для сборки ---
    ENV OPENSSL_STATIC=1
    ENV OPENSSL_LIB_DIR=/usr/lib
    ENV OPENSSL_INCLUDE_DIR=/usr/include
    ENV PROTOC=/usr/bin/protoc
    ENV RUSTFLAGS="-C target-feature=-crt-static"

    # --- Сборка RisingWave (занимает 30-60 мин, нужно ≥16GB RAM) ---
    RUN cargo build --release --bin risingwave 2>&1 | tail -20

    # --- Проверяем что бинарник собрался ---
    RUN ls -lh /build/target/release/risingwave && \\
        file /build/target/release/risingwave

    ########################################################
    # Stage 2: Минимальный рантайм на Alpine
    # Как в вашем k6 скрипте: чистый alpine + бинарник
    ########################################################
    FROM sberosc.sigma.sbrf.ru/docker.io/alpine:3.22.2

    RUN sed -i 's|https://dl-cdn.alpinelinux.org/alpine/|https://token:1234567890@sberosc.sigma.sbrf.ru/repo/apk/|g' /etc/apk/repositories

    # --- Только минимальные runtime-зависимости ---
    RUN apk add --no-cache --no-check-certificate \\
        ca-certificates \\
        libgcc \\
        libstdc++ \\
        openssl \\
        zlib \\
        lz4-libs \\
        zstd-libs \\
        snappy \\
        librdkafka

    # --- Копируем ТОЛЬКО бинарник из builder ---
    COPY --from=builder /build/target/release/risingwave /usr/bin/risingwave

    # --- Безопасность: non-root пользователь (как в вашем k6) ---
    RUN adduser -D -s /sbin/nologin rwuser

    # --- Каталоги для данных ---
    RUN mkdir -p /data /risingwave/state_store && \\
        chown -R rwuser:rwuser /data /risingwave

    # --- Удаляем ненужное для уменьшения attack surface ---
    RUN rm -rf /usr/share/doc /usr/share/man /var/cache/apk/*

    USER rwuser
    WORKDIR /home/rwuser

    EXPOSE 4566 5691

    ENTRYPOINT ["risingwave"]
    CMD ["single_node"]
    """

    sh "cat Dockerfile"

    echo "⏳ Сборка образа (займёт 30-60 минут, нужно ≥16GB RAM)..."
    sh """
        docker build \\
            --memory=16g \\
            --no-cache \\
            -t ${dockerRegistryURL}/${NexusGroupId}/${NexusArtifactId}:${RISINGWAVE_VERSION} \\
            .
    """

    echo "🔍 Проверяем что бинарник работает"
    sh "docker run --rm ${dockerRegistryURL}/${NexusGroupId}/${NexusArtifactId}:${RISINGWAVE_VERSION} --version"

    echo "📤 Пушим образ"
    sh "docker image push ${dockerRegistryURL}/${NexusGroupId}/${NexusArtifactId}:${RISINGWAVE_VERSION}"

    echo "✅ Образ ${ApplicationName}:${RISINGWAVE_VERSION} собран из исходников и запушен"
}
