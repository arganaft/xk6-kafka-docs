Для начала расскажу тебе об архитектуре сервиса и его предназначении, а потом о роли приложения в этой системе.
Ест сайт с дашбордами для трейдеров, которые должны обновляться в реальном времени. Дашборда создаются на основе LiveBook. LiveBook представляет собой таблицу из нескольких полей, в нем обьеденены данные из базы данных Postgresql, из сервиса по расчету рисков в реальном времени, сервиса по расчету доходности портфеля в реальном времени и других источников. LiveBook может создавать пользователь на сайте. Далее при создании дашбордов используют как основу один из LiveBook, берут из него необходимые поля, и может применить к ним группировки и агрегации, или просто запросить поля по какому либо фильтру. То есть делает запрос как к обычной таблице. Запрос он делает в виде JSON, и в ответ ждет данные по "подписке", в первоначальном ответе он получает полный объем данных, для полной отрисовки и заполнения дашборда, а потом только обновления (изменившиеся ячейки) из этого дашборда.

package com.websocket.trade.controller;

import com.fasterxml.jackson.databind.JsonNode;
import com.websocket.trade.service.TradeService;
import lombok.RequiredArgsConstructor;
import lombok.extern.slf4j.Slf4j;
import org.springframework.context.event.EventListener;
import org.springframework.messaging.handler.annotation.MessageMapping;
import org.springframework.messaging.handler.annotation.Payload;
import org.springframework.messaging.simp.SimpMessageHeaderAccessor;
import org.springframework.stereotype.Controller;
import org.springframework.web.socket.messaging.SessionDisconnectEvent;

import java.sql.SQLException;

@Controller
@RequiredArgsConstructor
@Slf4j
public class TradeClientController {

    private final TradeService tradeService;

    // Обработка отключения
    @EventListener
    public void handleWebSocketDisconnectListener(SessionDisconnectEvent event) {
        String sessionId = event.getSessionId();
        tradeService.removeSession(sessionId);
        log.info("Disconnected: {}", sessionId);
    }

    @MessageMapping("/config")
    public void handleClientRequest(@Payload JsonNode request, SimpMessageHeaderAccessor headerAccessor) throws SQLException {
        String sessionId = headerAccessor.getSessionId();

        String type = request.get("type").asText();
        if ("SetViewConfiguration".equals(type)) {
            JsonNode data = request.get("data");
            tradeService.addSubscription(data, sessionId);
        }
        if ("RequestImmediateData".equals(type)) {
            JsonNode data = request.get("data");
            tradeService.ImmediateData(data, sessionId);
        }
        if ("GetCurrentConfig".equals(type)) {
            String requestId = request.get("requestId").asText();
            tradeService.getCurrentConfig(requestId, sessionId);
        }
        if ("ClearAllConfigurations".equals(type)) {
            tradeService.removeSession(sessionId);
        }
        if ("RemoveConfigByRequestId".equals(type)) {
            String requestId = request.get("requestId").asText();
            tradeService.removeConfigByRequestId(requestId, sessionId);
        }

    }

}

ResultBroker тоже подписан на этот топик кафки, при поступлении нового плана запроса, он немедленно делает инициализирующий запрос в БД ClickHouse а также запрашивает данные из других источников, и отправляет эти данные в Redis по ключу sessionId, и потом по поступающим данным из топика кафки и других источников расчитывает данные в своей памяти, например фильтрации или агрегации, и отправляет только изменившиеся данные в Redis по ключу sessionId. package com.websocket.trade.service;

import com.fasterxml.jackson.databind.node.ArrayNode;

@Service
public class StreamViewConfigToSqlConverter {

    private static final String DEFAULT_TABLE = "livebook_FxSpotForwardTrade";

    public String convert(JsonNode config, String tableName) {
        if (config == null) {
            throw new IllegalArgumentException("Config cannot be null");
        }

        ArrayNode columns = getArrayNode(config, "columns");
        ArrayNode rowGroupColumns = getArrayNode(config, "rowGroupColumns");
        ArrayNode pivotColumns = getArrayNode(config, "pivotColumns");
        ArrayNode valueColumns = getArrayNode(config, "valueColumns");
        ArrayNode sortModel = getArrayNode(config, "sortModel");
        ArrayNode groupKeys = getArrayNode(config, "groupKeys");
        JsonNode filterModel = config.path("filterModel");
        int startRow = config.path("startRow").asInt(0);
        int endRow = config.path("endRow").asInt(100);

        // Собираем имена агрегированных полей (тех, что есть в valueColumns с aggFunc)
        Set<String> aggregatedFields = new HashSet<>();
        if (valueColumns != null) {
            for (JsonNode vc : valueColumns) {
                String fieldKey = vc.path("fieldKey").asText(null);
                String aggFunc = vc.path("aggFunc").asText(null);
                if (fieldKey != null && aggFunc != null) {
                    aggregatedFields.add(fieldKey);
                }
            }
        }

        boolean hasGrouping = rowGroupColumns != null && !rowGroupColumns.isEmpty();
        ArrayNode cleanGroupColumn = JsonNodeFactory.instance.arrayNode();
        Set<String> rgcFieldKeys = extractFieldKeySet(rowGroupColumns);
        Set<String> groopKeySet = extractFieldKeySet(groupKeys);
        rgcFieldKeys.removeAll(groopKeySet);
        if (rowGroupColumns != null) {
            for (JsonNode column : rowGroupColumns) {
                if (rgcFieldKeys.contains(column.get("fieldKey").asText())) {
                    cleanGroupColumn.add(column);
                }
            }
        }

        ArrayNode firstrowGroupColumn = JsonNodeFactory.instance.arrayNode();
        if (!cleanGroupColumn.isEmpty()) {
            firstrowGroupColumn.add(cleanGroupColumn.get(0));
        }
        if (pivotColumns instanceof ArrayNode && !pivotColumns.isEmpty()) {
            firstrowGroupColumn.addAll(pivotColumns);
        }

        // Реальные ключи группировки, которые будут в GROUP BY / SELECT
        Set<String> actualGroupByFields = new HashSet<>();
        for (JsonNode node : firstrowGroupColumn) {
            String fieldKey = node.path("fieldKey").asText(null);
            if (fieldKey != null) {
                actualGroupByFields.add(fieldKey);
            }
        }

        StringBuilder sql = new StringBuilder();

        // SELECT
        sql.append("SELECT ");
        String selectClause = buildSelectClause(columns, firstrowGroupColumn, valueColumns, hasGrouping, groupKeys);
        sql.append(selectClause);

        // FROM
        sql.append(" FROM ").append(tableName);

        // WHERE и HAVING (разделяем фильтры)
        FilterResult filterResult = buildFilters(filterModel, groupKeys, aggregatedFields);
        if (!filterResult.whereClause.isEmpty()) {
            sql.append(" WHERE ").append(filterResult.whereClause);
        }

        // GROUP BY
        if (!firstrowGroupColumn.isEmpty()) {
            String groupByClause = buildGroupByClause(firstrowGroupColumn, hasGrouping);
            if (!groupByClause.isEmpty()) {
                sql.append(" GROUP BY ").append(groupByClause);
            }
        }

        // HAVING
        if (!filterResult.havingClause.isEmpty()) {
            sql.append(" HAVING ").append(filterResult.havingClause);
        }

        String orderByClause = buildOrderByClause(sortModel, actualGroupByFields, valueColumns);
        if (!orderByClause.isEmpty()) {
            sql.append(" ORDER BY ").append(orderByClause);
        }

        // LIMIT / OFFSET
        int limit = endRow - startRow;
        if (limit > 0) {
            sql.append(" LIMIT ").append(limit);
        }
        if (startRow > 0) {
            sql.append(" OFFSET ").append(startRow);
        }

        return sql.toString();
    }

    // ==================== SELECT ====================

    private String buildSelectClause(
            ArrayNode columns,
            ArrayNode firstrowGroupColumn,
            ArrayNode valueColumns,
            boolean hasGrouping,
            ArrayNode groupKeys
    ) {
        List<String> selectParts = new ArrayList<>();
        if (!firstrowGroupColumn.isEmpty()) {
            List<String> groupByField = getGroupByFields(firstrowGroupColumn, true);
            selectParts.add(String.join(", ", groupByField));
            addAggregations(selectParts, valueColumns);
        } else if (hasGrouping && firstrowGroupColumn.isEmpty()) {
            addColumnsToSelect(selectParts, columns);
        } else {
            addColumnsToSelect(selectParts, columns);
        }
        return selectParts.isEmpty() ? "*" : String.join(", ", selectParts);
    }

    private void addColumnsToSelect(List<String> selectParts, ArrayNode columns) {
        if (columns == null || columns.isEmpty()) {
            selectParts.add("*");
            return;
        }
        for (JsonNode col : columns) {
            String fieldKey = getFieldKey(col);
            if (fieldKey != null) {
                selectParts.add(sanitizeIdentifier(fieldKey));
            }
        }
    }

    private void addAggregations(List<String> selectParts, ArrayNode valueColumns) {
        if (valueColumns == null) return;
        for (JsonNode valCol : valueColumns) {
            String aggFunc = getAggFunc(valCol);
            String fieldKey = getFieldKey(valCol);
            if (fieldKey != null && aggFunc != null) {
                String sanitized = sanitizeIdentifier(fieldKey);
                if ("unique_count".equals(aggFunc)) {
                    selectParts.add("COUNT(DISTINCT " + sanitized + ") AS " + sanitized);
                } else {
                    selectParts.add(aggFunc.toUpperCase() + "(" + sanitized + ") AS " + sanitized);
                }
            }
        }
    }

    // ==================== WHERE / HAVING ====================

    private static class FilterResult {
        String whereClause = "";
        String havingClause = "";
    }

    private FilterResult buildFilters(JsonNode filterModel, ArrayNode groupKeys, Set<String> aggregatedFields) {
        FilterResult result = new FilterResult();
        List<String> whereConditions = new ArrayList<>();
        List<String> havingConditions = new ArrayList<>();

        // Обработка filterModel
        if (filterModel != null && !filterModel.isMissingNode() && filterModel.isObject()) {
            Iterator<Map.Entry<String, JsonNode>> fields = filterModel.fields();
            while (fields.hasNext()) {
                Map.Entry<String, JsonNode> entry = fields.next();
                String fieldName = entry.getKey();
                JsonNode filterDef = entry.getValue();
                // Определяем, агрегат ли это поле
                boolean isAggregated = aggregatedFields.contains(fieldName);
                String filterSql = buildFilterCondition(fieldName, filterDef, true, isAggregated);
                if (filterSql != null && !filterSql.isEmpty()) {
                    if (isAggregated) {
                        havingConditions.add(filterSql);
                    } else {
                        whereConditions.add(filterSql);
                    }
                }
            }
        }

        // Обработка groupKeys (всегда в WHERE, так как это значения группировки)
        if (groupKeys != null && groupKeys.size() > 0) {
            List<JsonNode> timeGroupKeys = groupKeys.findParents("groupBy");
            Map<String, JsonNode> fieldKeyMap = timeGroupKeys.stream()
                    .filter(node -> node.has("fieldKey"))
                    .collect(Collectors.toMap(
                            node -> node.get("fieldKey").asText(),
                            node -> node,
                            (existing, replacement) -> existing
                    ));
            for (JsonNode node : groupKeys) {
                if (node.has("fieldKey") && node.has("value")) {
                    String fieldKey = node.get("fieldKey").asText();
                    String value = node.get("value").asText();
                    if (fieldKeyMap.containsKey(fieldKey)) {
                        String groupBy = fieldKeyMap.get(fieldKey).get("groupBy").asText();
                        whereConditions.add(String.format("EXTRACT(%s FROM %s) = %s",
                                groupBy, sanitizeIdentifier(fieldKey), escapeValue(value)));
                    } else {
                        String escapedValue = node.get("value").isNumber() ?
                                value : "'" + getTimestampOrString(value) + "'";
                        whereConditions.add(sanitizeIdentifier(fieldKey) + " = " + escapedValue);
                    }
                }
            }
        }

        result.whereClause = String.join(" AND ", whereConditions);
        result.havingClause = String.join(" AND ", havingConditions);
        return result;
    }

    private String buildFilterCondition(String fieldName, JsonNode filterDef,
                                        boolean sanitizeEnabled, boolean isAggregated) {
        if (filterDef == null || filterDef.isMissingNode()) return null;

        String filterType = filterDef.path("filterType").asText("");
        // Для агрегированных полей в HAVING используем тот же алиас, что и в SELECT
        String sanitizedField = isAggregated ? sanitizeIdentifier(fieldName) : sanitizeIdentifier(fieldName);

        if (filterDef.has("operator")) {
            String operator = filterDef.path("operator").asText("AND");
            String cond1 = buildSingleFilterCondition(sanitizedField, filterDef.get("condition1"), filterType, isAggregated);
            String cond2 = buildSingleFilterCondition(sanitizedField, filterDef.get("condition2"), filterType, isAggregated);
            List<String> parts = new ArrayList<>();
            if (cond1 != null) parts.add(cond1);
            if (cond2 != null) parts.add(cond2);
            if (parts.isEmpty()) return null;
            if (parts.size() == 1) return parts.get(0);
            return "(" + String.join(" OR ".equalsIgnoreCase(operator) ? " OR " : " AND ", parts) + ")";
        }

        if ("set".equalsIgnoreCase(filterType)) {
            return buildSetFilter(sanitizedField, filterDef);
        }

        return buildSingleFilterCondition(sanitizedField, filterDef, filterType, isAggregated);
    }

    private String buildSingleFilterCondition(String sanitizedField, JsonNode condition,
                                              String filterType, boolean isAggregated) {
        if (condition == null || condition.isMissingNode()) return null;
        String type = condition.path("type").asText("");

        switch (type) {
            case "equals":
                return sanitizedField + " = " + formatValue(extractFilterValue(condition, filterType), filterType);
            case "notEqual":
                return sanitizedField + " != " + formatValue(extractFilterValue(condition, filterType), filterType);
            case "contains":
                return sanitizedField + " LIKE " + escapeValue("%" + condition.path("filter").asText() + "%");
            case "notContains":
                return sanitizedField + " NOT LIKE " + escapeValue("%" + condition.path("filter").asText() + "%");
            case "startsWith":
                return sanitizedField + " LIKE " + escapeValue(condition.path("filter").asText() + "%");
            case "endsWith":
                return sanitizedField + " LIKE " + escapeValue("%" + condition.path("filter").asText());
            case "lessThan":
                return sanitizedField + " < " + formatValue(extractFilterValue(condition, filterType), filterType);
            case "lessThanOrEqual":
                return sanitizedField + " <= " + formatValue(extractFilterValue(condition, filterType), filterType);
            case "greaterThan":
                return sanitizedField + " > " + formatValue(extractFilterValue(condition, filterType), filterType);
            case "greaterThanOrEqual":
                return sanitizedField + " >= " + formatValue(extractFilterValue(condition, filterType), filterType);
            case "inRange":
                if ("number".equalsIgnoreCase(filterType)) {
                    // Для чисел – без кавычек
                    String start = condition.path("filter").asText();
                    String end = condition.path("filterTo").asText();
                    // Если значения не заданы, используем 0
                    if (start.isEmpty()) start = "0";
                    if (end.isEmpty()) end = "0";
                    return sanitizedField + " BETWEEN " + start + " AND " + end;
                } else {
                    String start = escapeValue(condition.path("filter").asText(""));
                    String end = escapeValue(condition.path("filterTo").asText(""));
                    return sanitizedField + " BETWEEN " + start + " AND " + end;
                }
            case "blank":
            case "empty":
                return "(" + sanitizedField + " IS NULL)";
            case "notBlank":
            case "notEmpty":
                return "(" + sanitizedField + " IS NOT NULL)";
            default:
                if (condition.has("filter")) {
                    return sanitizedField + " = " + formatValue(extractFilterValue(condition, filterType), filterType);
                }
                return null;
        }
    }

    // Обёртка для чисел (если filterType=number, не добавляем кавычки)
    private String formatValue(String value, String filterType) {
        if ("number".equalsIgnoreCase(filterType)) {
            return value; // уже число в виде строки без кавычек
        }
        return escapeValue(value);
    }

    private String extractFilterValue(JsonNode condition, String filterType) {
        if ("number".equalsIgnoreCase(filterType)) {
            if (condition.path("filter").isNumber()) {
                return String.valueOf(condition.path("filter").numberValue());
            }
            return condition.path("filter").asText("0");
        }
        if ("date".equalsIgnoreCase(filterType)) {
            String dateFrom = condition.path("dateFrom").asText(null);
            if (dateFrom != null) {
                return dateFrom;
            }
        }
        return condition.path("filter").asText("");
    }

    private String buildSetFilter(String sanitizedField, JsonNode filterDef) {
        ArrayNode values = getArrayNode(filterDef, "values");
        if (values == null || values.isEmpty()) {
            return sanitizedField + " IN (NULL)";
        }
        StringJoiner joiner = new StringJoiner(", ");
        boolean hasNull = false;
        for (JsonNode val : values) {
            if (val.isNull()) {
                hasNull = true;
            } else {
                joiner.add(escapeValue(val.asText()));
            }
        }
        if (hasNull && joiner.length() > 0) {
            return "(" + sanitizedField + " IN (" + joiner + ") OR " + sanitizedField + " IS NULL)";
        } else if (hasNull) {
            return sanitizedField + " IS NULL";
        } else {
            return sanitizedField + " IN (" + joiner + ")";
        }
    }

    // ==================== GROUP BY ====================

    private String buildGroupByClause(ArrayNode rowGroupColumns, boolean hasGrouping) {
        if (!hasGrouping) return "";
        List<String> groupByFields = getGroupByFields(rowGroupColumns, false);
        return String.join(", ", groupByFields);
    }

    private List<String> getGroupByFields(ArrayNode rowGroupColumns, boolean addAlias) {
        List<String> groupByFields = new ArrayList<>();
        for (JsonNode node : rowGroupColumns) {
            String dateGroup = getDateGroup(node);
            String fieldKey = sanitizeIdentifier(node.path("fieldKey").asText());
            if (node.has("fieldKey") && dateGroup == null) {
                groupByFields.add(addAlias ? fieldKey + " AS " + fieldKey : fieldKey);
            }
            if (node.has("fieldKey") && dateGroup != null) {
                String dateExpr = "EXTRACT(" + dateGroup.toUpperCase() +
                        " FROM " + sanitizeIdentifier(node.path("fieldKey").asText()) + ")";
                groupByFields.add(addAlias ? dateExpr + " AS " + fieldKey : dateExpr);
            }
        }
        return groupByFields;
    }

    private String getDateGroup(JsonNode node) {
        if (node == null) return null;
        String dateGroup = node.path("groupBy").asText(null);
        if (dateGroup == null) return null;
        return switch (dateGroup.toLowerCase()) {
            case "year", "month", "day" -> dateGroup.toLowerCase();
            default -> null;
        };
    }

    // ==================== ORDER BY ====================

    private String buildOrderByClause(ArrayNode sortModel, Set<String> actualGroupByFields,
                                      ArrayNode valueColumns) {
        if (sortModel == null || sortModel.isEmpty()) return "";
        List<String> valueFields = (valueColumns != null)
                ? valueColumns.findValuesAsText("fieldKey")
                : Collections.emptyList();
        List<String> orderParts = new ArrayList<>();
        for (JsonNode sortItem : sortModel) {
            String colId = sortItem.path("colId").asText(null);
            String sort = sortItem.path("sort").asText("asc");
            if (colId != null && !colId.isEmpty()
                    && isContainsGroupFieldOrAggregation(colId, actualGroupByFields, valueFields)) {
                orderParts.add(sanitizeIdentifier(colId) + " " + ("desc".equalsIgnoreCase(sort) ? "DESC" : "ASC"));
            }
        }
        return String.join(", ", orderParts);
    }

    private boolean isContainsGroupFieldOrAggregation(String colId, Set<String> groupByFields,
                                                      List<String> valueFields) {
        if (groupByFields.isEmpty() && valueFields.isEmpty()) return true;
        return groupByFields.contains(colId) || valueFields.contains(colId);
    }

    // ==================== Утилиты ====================

    public Set<String> extractFieldKeySet(ArrayNode arrayNode) {
        Set<String> result = new HashSet<>();
        if (arrayNode == null) return result;
        for (JsonNode node : arrayNode) {
            JsonNode field = node.get("fieldKey");
            if (field != null) result.add(field.asText());
        }
        return result;
    }

    private String sanitizeIdentifier(String identifier) {
        if (identifier == null || identifier.isEmpty())
            throw new IllegalArgumentException("Identifier cannot be null or empty");
        String sanitized = identifier.replaceAll("[^a-zA-Z0-9_.]", "");
        if (sanitized.isEmpty())
            throw new IllegalArgumentException("Invalid identifier after sanitization: " + identifier);
        if (sanitized.contains(".")) {
            StringJoiner joiner = new StringJoiner(".");
            for (String part : sanitized.split("\\.")) {
                joiner.add("\"" + part + "\"");
            }
            return joiner.toString();
        }
        return "\"" + sanitized + "\"";
    }

    private String escapeValue(String value) {
        if (value == null) return "NULL";
        String escaped = value.replace("'", "''").replace("\\", "\\\\");
        return "'" + escaped + "'";
    }

    private String getFieldKey(JsonNode column) {
        return column != null ? column.path("fieldKey").asText(null) : null;
    }

    private String getAggFunc(JsonNode column) {
        if (column == null) return null;
        String aggFunc = column.path("aggFunc").asText(null);
        if (aggFunc == null) return null;
        return switch (aggFunc.toLowerCase()) {
            case "sum", "avg", "min", "max", "count", "unique_count" -> aggFunc.toLowerCase();
            default -> null;
        };
    }

    private ArrayNode getArrayNode(JsonNode parent, String fieldName) {
        JsonNode node = parent.path(fieldName);
        return node.isArray() ? (ArrayNode) node : null;
    }

    private String getTimestampOrString(Object value) {
        if (value instanceof Timestamp ts) return String.valueOf(ts.getTime());
        if (value instanceof Date date) return String.valueOf(date.getTime());
        if (value instanceof Instant instant) return String.valueOf(Timestamp.from(instant).getTime());
        if (value instanceof LocalDateTime ldt)
            return String.valueOf(Timestamp.from(ldt.atZone(ZoneId.systemDefault()).toInstant()).getTime());
        if (value instanceof ZonedDateTime zdt)
            return String.valueOf(Timestamp.from(zdt.toInstant()).getTime());
        if (value instanceof Long epochMs) return String.valueOf(epochMs);
        return value.toString();
    }
}
