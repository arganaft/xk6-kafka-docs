package com.yourapp.livebook.dto;

import io.swagger.v3.oas.annotations.media.Schema;
import jakarta.validation.constraints.Min;
import jakarta.validation.constraints.NotBlank;
import jakarta.validation.constraints.NotEmpty;
import jakarta.validation.constraints.NotNull;
import lombok.AllArgsConstructor;
import lombok.Builder;
import lombok.Data;
import lombok.NoArgsConstructor;

import java.util.List;

@Data
@Builder
@NoArgsConstructor
@AllArgsConstructor
@Schema(description = "Описание LiveBook")
public class LiveBookDto {

    @NotNull
    @Min(1)
    @Schema(
            description = "Идентификатор LiveBook",
            example = "110"
    )
    private Long liveBookId;

    @NotBlank
    @Schema(
            description = "Название LiveBook",
            example = "LiveBook110"
    )
    private String liveBookName;

    @NotBlank
    @Schema(
            description = "Фильтр LiveBook",
            example = "Trade.portfolioId==6052"
    )
    private String filter;

    @NotEmpty
    @Schema(
            description = "Список полей LiveBook",
            example = """
                    [
                      "Entity.id",
                      "FxSpotForwardTrade.baseAmount",
                      "FxSpotForwardTrade.baseCurrency.code",
                      "Trade.portfolioId"
                    ]
                    """
    )
    private List<@NotBlank String> fields;
}

## 4. Контроллер

Файл: `src/main/java/com/yourapp/livebook/controller/LiveBookController.java`

```java
package com.yourapp.livebook.controller;

import com.yourapp.livebook.controller.api.LiveBookControllerApi;
import com.yourapp.livebook.dto.LiveBookCreateResponse;
import com.yourapp.livebook.dto.LiveBookDto;
import com.yourapp.livebook.exception.LiveBookAlreadyExistsException;
import com.yourapp.livebook.service.LiveBookService;
import io.micrometer.core.instrument.Counter;
import io.micrometer.core.instrument.MeterRegistry;
import io.micrometer.core.instrument.Timer;
import io.swagger.v3.oas.annotations.tags.Tag;
import jakarta.validation.Valid;
import lombok.extern.slf4j.Slf4j;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.http.HttpStatus;
import org.springframework.http.MediaType;
import org.springframework.http.ResponseEntity;
import org.springframework.web.bind.annotation.*;

import java.util.List;

@Slf4j
@RestController
@RequestMapping("/api/v1/livebook")
@Tag(name = "LiveBook API", description = "API для управления LiveBook")
public class LiveBookController implements LiveBookControllerApi {

    private final LiveBookService liveBookService;

    private final Counter liveBookCreateCounter;
    private final Counter liveBookCreateErrorCounter;
    private final Timer liveBookCreateRequestTimer;

    private final Counter liveBookGetAllCounter;
    private final Counter liveBookGetAllErrorCounter;
    private final Timer liveBookGetAllRequestTimer;

    @Autowired
    public LiveBookController(
            LiveBookService liveBookService,
            MeterRegistry meterRegistry
    ) {
        this.liveBookService = liveBookService;

        this.liveBookCreateCounter = meterRegistry.counter("livebook_create_total");
        this.liveBookCreateErrorCounter = meterRegistry.counter("livebook_create_error_total");
        this.liveBookCreateRequestTimer = meterRegistry.timer("livebook_create_duration_seconds");

        this.liveBookGetAllCounter = meterRegistry.counter("livebook_get_all_total");
        this.liveBookGetAllErrorCounter = meterRegistry.counter("livebook_get_all_error_total");
        this.liveBookGetAllRequestTimer = meterRegistry.timer("livebook_get_all_duration_seconds");
    }

    @Override
    @PostMapping(
            value = "/create",
            consumes = MediaType.APPLICATION_JSON_VALUE,
            produces = MediaType.APPLICATION_JSON_VALUE
    )
    public ResponseEntity<LiveBookCreateResponse> createLiveBook(
            @Valid @RequestBody LiveBookDto request
    ) {
        Timer.Sample sample = Timer.start();
        liveBookCreateCounter.increment();

        try {
            LiveBookDto createdLiveBook = liveBookService.create(request);

            log.info(
                    "LiveBook created: id={}, name={}",
                    createdLiveBook.getLiveBookId(),
                    createdLiveBook.getLiveBookName()
            );

            LiveBookCreateResponse response = LiveBookCreateResponse.builder()
                    .code(HttpStatus.OK.value())
                    .result("Live book created")
                    .error(null)
                    .id(createdLiveBook.getLiveBookId())
                    .build();

            return ResponseEntity.ok(response);
        } catch (LiveBookAlreadyExistsException e) {
            liveBookCreateErrorCounter.increment();

            log.warn("LiveBook already exists: id={}", request.getLiveBookId(), e);

            LiveBookCreateResponse response = LiveBookCreateResponse.builder()
                    .code(HttpStatus.CONFLICT.value())
                    .result(null)
                    .error(e.getMessage())
                    .id(request.getLiveBookId())
                    .build();

            return ResponseEntity.status(HttpStatus.CONFLICT).body(response);
        } catch (Exception e) {
            liveBookCreateErrorCounter.increment();

            log.error("Error while creating LiveBook: id={}", request.getLiveBookId(), e);

            LiveBookCreateResponse response = LiveBookCreateResponse.builder()
                    .code(HttpStatus.INTERNAL_SERVER_ERROR.value())
                    .result(null)
                    .error("Internal server error")
                    .id(request.getLiveBookId())
                    .build();

            return ResponseEntity.status(HttpStatus.INTERNAL_SERVER_ERROR).body(response);
        } finally {
            sample.stop(liveBookCreateRequestTimer);
        }
    }

    @Override
    @GetMapping(
            value = "/all",
            produces = MediaType.APPLICATION_JSON_VALUE
    )
    public ResponseEntity<List<LiveBookDto>> getAllLiveBooks() {
        Timer.Sample sample = Timer.start();
        liveBookGetAllCounter.increment();

        try {
            List<LiveBookDto> liveBooks = liveBookService.findAll();

            log.info("Fetching all LiveBooks: count={}", liveBooks.size());

            return ResponseEntity.ok(liveBooks);
        } catch (Exception e) {
            liveBookGetAllErrorCounter.increment();

            log.error("Error while fetching all LiveBooks", e);

            return ResponseEntity.status(HttpStatus.INTERNAL_SERVER_ERROR).build();
        } finally {
            sample.stop(liveBookGetAllRequestTimer);
        }
    }
}
```

## 6. Простая in-memory реализация сервиса

Если у тебя уже есть БД, эту реализацию можно заменить на работу с `Repository`.

Файл: `src/main/java/com/yourapp/livebook/service/impl/InMemoryLiveBookService.java`

```java
package com.yourapp.livebook.service.impl;

import com.yourapp.livebook.dto.LiveBookDto;
import com.yourapp.livebook.exception.LiveBookAlreadyExistsException;
import com.yourapp.livebook.service.LiveBookService;
import org.springframework.stereotype.Service;

import java.util.ArrayList;
import java.util.Comparator;
import java.util.List;
import java.util.Map;
import java.util.concurrent.ConcurrentHashMap;

@Service
public class InMemoryLiveBookService implements LiveBookService {

    private final Map<Long, LiveBookDto> storage = new ConcurrentHashMap<>();

    @Override
    public LiveBookDto create(LiveBookDto request) {
        LiveBookDto liveBook = copy(request);

        LiveBookDto previous = storage.putIfAbsent(liveBook.getLiveBookId(), liveBook);

        if (previous != null) {
            throw new LiveBookAlreadyExistsException(
                    "LiveBook with id " + liveBook.getLiveBookId() + " already exists"
            );
        }

        return copy(liveBook);
    }

    @Override
    public List<LiveBookDto> findAll() {
        return storage.values()
                .stream()
                .map(this::copy)
                .sorted(Comparator.comparing(LiveBookDto::getLiveBookId))
                .toList();
    }

    private LiveBookDto copy(LiveBookDto source) {
        return LiveBookDto.builder()
                .liveBookId(source.getLiveBookId())
                .liveBookName(source.getLiveBookName())
                .filter(source.getFilter())
                .fields(source.getFields() == null ? List.of() : new ArrayList<>(source.getFields()))
                .build();
    }
}
```

<details>
<summary>Если нужно хранить LiveBook в БД</summary>

Тогда вместо `InMemoryLiveBookService` лучше сделать:

- `LiveBookEntity`
- `LiveBookRepository extends JpaRepository<LiveBookEntity, Long>`
- `LiveBookServiceImpl`, который мапит `LiveBookDto` в entity и обратно.

Но так как в примере нет требований к БД, для добавления REST-ручек достаточно показанного варианта.

</details>
