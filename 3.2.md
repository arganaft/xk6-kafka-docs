FROM sberosc.sigma.sbrf.ru/docker.io/alpine:3.22.2

RUN sed -i 's|https://dl-cdn.alpinelinux.org/alpine/|https://token:1234567890@sberosc.sigma.sbrf.ru/repo/apk/|g' /etc/apk/repositories

RUN apk add --no-cache --no-check-certificate \
    ca-certificates \
    gcompat \
    libgcc \
    libstdc++ \
    openssl

COPY --from=source /risingwave/bin/risingwave /usr/local/bin/risingwave

RUN adduser -D -s /sbin/nologin rwuser
RUN mkdir -p /data && chown rwuser:rwuser /data

USER rwuser
WORKDIR /home/rwuser

EXPOSE 4566 5691

ENTRYPOINT ["risingwave"]
CMD ["single_node"]
