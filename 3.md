import { check } from 'k6';
import http from 'k6/http';
import { group } from 'k6';

// ==================== НАСТРОЙКИ ====================
const TRADES_PER_ITERATION = 50; // Количество сделок за одну итерацию VU
const OBJECTS_PER_TRADE = 4;     // 1 Contract + 1 Trade + 2 Cashflow

const BASE_URL_GLOBAL_ID = 'http://object-repository-global-id-generation-service-k8s.quantara-dev.delta.sbrf.ru';
const BASE_URL_ID = 'http://object-repository-id-generation-service-k8s.quantara-dev.delta.sbrf.ru';
const BASE_URL_TXN = 'http://object-repository-transaction-service-k8s.quantara-dev.delta.sbrf.ru';
const BASE_URL_LOCK = 'http://object-repository-lock-service-k8s.quantara-dev.delta.sbrf.ru';

// ==================== КОНФИГУРАЦИЯ ТЕСТА ====================
export const options = {
  scenarios: {
    growing_scenario: {
      executor: "ramping-vus",
      startVUs: 10,
      stages: [
        { duration: '1m', target: 100 },
        { duration: '10m', target: 100 },
      ],
    }
  },
  thresholds: {
    http_req_failed: ['rate<0.001'],
    http_req_duration: ['p(95)<500'],
  },
};

// ==================== ГЕНЕРАТОРЫ PAYLOAD ====================

// ==================== ОСНОВНОЙ СЦЕНАРИЙ ====================

export default function () {

  const headers = {
    'Content-Type': 'application/json',
    'Accept': 'application/json'
  };

  const processId = crypto.randomUUID();

  // Общее количество ID объектов: на каждую сделку нужно 4 (contract, trade, cashflow1, cashflow2)
  const totalObjectIds = TRADES_PER_ITERATION * OBJECTS_PER_TRADE;

  // ------------------------------------------------------------------
  // 1. Batch-получение global ID (по одному на каждую сделку)
  // ------------------------------------------------------------------
  let globalIdStart;

  group('batch-global-id-generation', function () {
    const params = {
      headers: headers,
      tags: { name: 'object-repository-global-id-generation-service-batch' },
    };

    const res = http.get(
      `${BASE_URL_GLOBAL_ID}/api/v1/global-id/batch/${TRADES_PER_ITERATION}`,
      params
    );
    checkResponse('global-id-batch', res);
    check(res, { 'global-id batch status 200': (r) => r.status === 200 });

    globalIdStart = res.json('startId');
  });

  // ------------------------------------------------------------------
  // 2. Batch-получение object ID (contract + trade + 2 cashflow на сделку)
  // ------------------------------------------------------------------
  let objectIdStart;

  group('batch-id-generation', function () {
    const params = {
      headers: headers,
      tags: { name: 'object-repository-id-generation-service-batch' },
    };

    const res = http.get(
      `${BASE_URL_ID}/api/v1/id/batch/${totalObjectIds}`,
      params
    );
    checkResponse('id-batch', res);
    check(res, { 'id batch status 200': (r) => r.status === 200 });

    objectIdStart = res.json('startId');
  });

  // ------------------------------------------------------------------
  // 3. Открытие транзакции
  // ------------------------------------------------------------------
  let trnId;

  group('transaction-begin', function () {
    const params = {
      headers: headers,
      tags: { name: 'object-repository-transaction-service - begin' },
    };

    const res = http.get(
      `${BASE_URL_TXN}/api/transactions/begin?processId=${processId}`,
      params
    );
    checkResponse('transaction-begin', res);
    check(res, { 'begin status 200': (r) => r.status === 200 });

    trnId = res.json('result');
  });

  // ------------------------------------------------------------------
  // 4. Загрузка сделок (каждая = lock → 4 save → release)
  // ------------------------------------------------------------------

  for (let i = 0; i < TRADES_PER_ITERATION; i++) {

    const globalId = globalIdStart + i;

    // Смещение внутри пула object ID: 4 объекта на сделку
    const base = objectIdStart + i * OBJECTS_PER_TRADE;
    const contractId = base;
    const tradeId    = base + 1;
    const cashFlowId1 = base + 2;
    const cashFlowId2 = base + 3;

    const lockId = crypto.randomUUID();

    // ---------- Acquire lock ----------
    group('lock-acquire', function () {
      const params = {
        headers: headers,
        tags: { name: 'object-repository-lock-service - acquire' },
      };
      const payload = JSON.stringify({ lockId: lockId, ttl: 1200000 });

      const res = http.post(
        `${BASE_URL_LOCK}/api/locks/${globalId}/acquire`,
        payload,
        params
      );
      checkResponse('lock-acquire', res);
      check(res, { 'lock acquire 200': (r) => r.status === 200 });
    });

    // ---------- Save Contract ----------
    group('save-contract', function () {
      const params = {
        headers: {
          'Content-Type': 'application/json',
          'Accept': 'application/json',
          'Object-Type': 'trades-Contract',
          'Global-Id': `${globalId}`,
          'Transaction-Id': `${trnId}`,
          'Lock-Id': `${lockId}`,
          'Id': `${contractId}`
        },
        tags: { name: 'object-repository-transaction-service - save' },
      };

      const res = http.post(
        `${BASE_URL_TXN}/api/transactions/save`,
        generateContract(globalId),
        params
      );
      checkResponse('save-contract', res);
      check(res, { 'save contract 200': (r) => r.status === 200 });
    });

    // ---------- Save Trade ----------
    group('save-trade', function () {
      const params = {
        headers: {
          'Content-Type': 'application/json',
          'Accept': 'application/json',
          'Object-Type': 'trades-fx-FxSpotForward',
          'Global-Id': `${globalId}`,
          'Transaction-Id': `${trnId}`,
          'Lock-Id': `${lockId}`,
          'Id': `${tradeId}`,
          'Actual-From': '2026-02-17',
          'Draft-Status': 'CONFIRMED',
          'Revision': '1',
          'version': '1'
        },
        tags: { name: 'object-repository-transaction-service - save' },
      };

      const res = http.post(
        `${BASE_URL_TXN}/api/transactions/save`,
        generateTrade(globalId, tradeId),
        params
      );
      checkResponse('save-trade', res);
      check(res, { 'save trade 200': (r) => r.status === 200 });
    });

    // ---------- Save Cashflow 1 ----------
    group('save-cashflow-1', function () {
      const params = {
        headers: {
          'Content-Type': 'application/json',
          'Accept': 'application/json',
          'Object-Type': 'trades-Cashflow',
          'Global-Id': `${globalId}`,
          'Transaction-Id': `${trnId}`,
          'Lock-Id': `${lockId}`,
          'Id': `${cashFlowId1}`,
          'Parent-Id': `${tradeId}`
        },
        tags: { name: 'object-repository-transaction-service - save' },
      };

      const res = http.post(
        `${BASE_URL_TXN}/api/transactions/save`,
        generateCashflow(cashFlowId1, tradeId),
        params
      );
      checkResponse('save-cashflow-1', res);
      check(res, { 'save cashflow-1 200': (r) => r.status === 200 });
    });

    // ---------- Save Cashflow 2 ----------
    group('save-cashflow-2', function () {
      const params = {
        headers: {
          'Content-Type': 'application/json',
          'Accept': 'application/json',
          'Object-Type': 'trades-Cashflow',
          'Global-Id': `${globalId}`,
          'Transaction-Id': `${trnId}`,
          'Lock-Id': `${lockId}`,
          'Id': `${cashFlowId2}`,
          'Parent-Id': `${tradeId}`
        },
        tags: { name: 'object-repository-transaction-service - save' },
      };

      const res = http.post(
        `${BASE_URL_TXN}/api/transactions/save`,
        generateCashflow(cashFlowId2, tradeId),
        params
      );
      checkResponse('save-cashflow-2', res);
      check(res, { 'save cashflow-2 200': (r) => r.status === 200 });
    });

    // ---------- Release lock ----------
    group('lock-release', function () {
      const params = {
        headers: headers,
        tags: { name: 'object-repository-lock-service - release' },
      };
      const payload = JSON.stringify({ lockId: lockId });

      const res = http.del(
        `${BASE_URL_LOCK}/api/locks/${globalId}/release/${lockId}`,
        payload,
        params
      );
      checkResponse('lock-release', res);
      check(res, { 'lock release 200': (r) => r.status === 200 });
    });
  }

  // ------------------------------------------------------------------
  // 5. Commit транзакции (один раз после всех сделок)
  // ------------------------------------------------------------------
  group('transaction-commit', function () {
    const params = {
      headers: headers,
      tags: { name: 'object-repository-transaction-service - commit' },
    };

    const res = http.get(
      `${BASE_URL_TXN}/api/transactions/commit/${trnId}/${processId}`,
      params
    );
    checkResponse('transaction-commit', res);
    check(res, { 'commit status 200': (r) => r.status === 200 });
  });
}

## Что изменилось по сравнению с оригиналом

| Аспект | Было | Стало |
|---|---|---|
| **Global ID** | 1 запрос → 1 id | 1 batch-запрос → `TRADES_PER_ITERATION` id за раз |
| **Object ID** | 4 отдельных запроса | 1 batch-запрос → `TRADES_PER_ITERATION × 4` id за раз |
| **Количество сделок** | 1 сделка на итерацию | `TRADES_PER_ITERATION` сделок (настраивается) |
| **Транзакция** | open → 4 save → commit | open → *N × 4* save → **один** commit |
| **Lock** | 1 acquire / 1 release | По одному acquire/release **на каждую сделку** (lock привязан к `globalId`) |

### Логика распределения ID

```
batch global-id:  startId .. startId + N-1        (по 1 на сделку)
batch object-id:  startId .. startId + N*4 - 1    (по 4 на сделку)

Для сделки i:
  globalId    = globalIdStart + i
  contractId  = objectIdStart + i*4
  tradeId     = objectIdStart + i*4 + 1
  cashflowId1 = objectIdStart + i*4 + 2
  cashflowId2 = objectIdStart + i*4 + 3
```

Чтобы изменить количество сделок в одной транзакции — поменяйте константу `TRADES_PER_ITERATION` в начале файла.
