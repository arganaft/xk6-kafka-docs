function retryDelay(attempt) {
  const delay = Math.min(
    RETRY_BASE_DELAY_MS * Math.pow(2, attempt - 1) + Math.random() * 200,
    RETRY_MAX_DELAY_MS
  );
  sleep(delay / 1000);
}

function getWithRetry(url, params, serviceName) {
  let lastResponse = null;

  for (let attempt = 1; attempt <= MAX_RETRIES; attempt++) {
    try {
      lastResponse = http.get(url, params);
      if (lastResponse && lastResponse.status === 200) {
        return lastResponse;
      }
      console.warn(
        `[${serviceName}] attempt ${attempt}/${MAX_RETRIES} — ` +
        `status: ${lastResponse ? lastResponse.status : 'null'}, ` +
        `body: ${lastResponse ? lastResponse.body : 'N/A'}`
      );
    } catch (e) {
      console.warn(
        `[${serviceName}] attempt ${attempt}/${MAX_RETRIES} — exception: ${e}`
      );
    }

    if (attempt < MAX_RETRIES) retryDelay(attempt);
  }

  console.error(`[${serviceName}] all ${MAX_RETRIES} attempts exhausted — URL: ${url}`);
  return lastResponse;
}

export default function () {

  const headers = {
    'Content-Type': 'application/json',
    'Accept': 'application/json'
  };

  const processId = crypto.randomUUID();
  const totalObjectIds = TRADES_PER_ITERATION * OBJECTS_PER_TRADE;

  // Массив для хранения всех блокировок, которые нужно освободить после commit
  const acquiredLocks = [];

  // ------------------------------------------------------------------
  // 1. Batch-получение global ID
  // ------------------------------------------------------------------
  let globalIdStart;

  group('batch-global-id-generation', function () {
    const params = {
      headers: headers,
      tags: { name: 'global-id-generation-service-batch' },
    };

    const res = getWithRetry(
      `${BASE_URL_GLOBAL_ID}/api/v1/global-id/batch/${TRADES_PER_ITERATION}`,
      params,
      'global-id-batch'
    );

    const ok = check(res, {
      'global-id batch: response received': (r) => r !== null,
      'global-id batch: status 200': (r) => r && r.status === 200,
      'global-id batch: startId present': (r) => r && r.json('startId') !== undefined,
    });

    if (ok && res && res.status === 200) {
      globalIdStart = res.json('startId');
    }
  });

  if (!globalIdStart) {
    console.error('FATAL: не удалось получить batch global-id, прерываем итерацию');
    return;
  }

  // ------------------------------------------------------------------
  // 2. Batch-получение object ID
  // ------------------------------------------------------------------
  let objectIdStart;

  group('batch-id-generation', function () {
    const params = {
      headers: headers,
      tags: { name: 'id-generation-service-batch' },
    };

    const res = getWithRetry(
      `${BASE_URL_ID}/api/v1/id/batch/${totalObjectIds}`,
      params,
      'id-batch'
    );

    const ok = check(res, {
      'id batch: response received': (r) => r !== null,
      'id batch: status 200': (r) => r && r.status === 200,
      'id batch: startId present': (r) => r && r.json('startId') !== undefined,
    });

    if (ok && res && res.status === 200) {
      objectIdStart = res.json('startId');
    }
  });

  if (!objectIdStart) {
    console.error('FATAL: не удалось получить batch object-id, прерываем итерацию');
    return;
  }

  // ------------------------------------------------------------------
  // 3. Открытие транзакции
  // ------------------------------------------------------------------
  let trnId;

  group('transaction-begin', function () {

      headers: headers,
      tags: { name: 'transaction-service - begin' },
    };

    const res = getWithRetry(
      `${BASE_URL_TXN}/api/transactions/begin?processId=${processId}`,
      params,
      'transaction-begin'
    );

    const ok = check(res, {
      'begin: response received': (r) => r !== null,
      'begin: status 200': (r) => r && r.status === 200,
    });

    if (ok && res && res.status === 200) {
      trnId = res.json('result');
    }
  });

  if (!trnId) {
    console.error('FATAL: не удалось открыть транзакцию, прерываем итерацию');
    return;
  }

  // ------------------------------------------------------------------
  // 4. Acquire locks + Save объектов (без release!)
  // ------------------------------------------------------------------

  let allSavesOk = true;

  for (let i = 0; i < TRADES_PER_ITERATION; i++) {

    const globalId = globalIdStart + i;

    const base = objectIdStart + i * OBJECTS_PER_TRADE;
    const contractId  = base;
    const tradeId     = base + 1;
    const cashFlowId1 = base + 2;
    const cashFlowId2 = base + 3;

    const lockId = crypto.randomUUID();

    // ---------- Acquire lock ----------
    group('lock-acquire', function () {
      const params = {
        headers: headers,
        tags: { name: 'lock-service - acquire' },
      };
      const payload = JSON.stringify({ lockId: lockId, ttl: 1200000 });

        `${BASE_URL_LOCK}/api/locks/${globalId}/acquire`,
        payload,
        params,
        'lock-acquire'

        'lock acquire: status 200': (r) => r && r.status === 200,
      });

      if (ok && res && res.status === 200) {
        // Запоминаем блокировку для освобождения после commit
        acquiredLocks.push({ globalId: globalId, lockId: lockId });
      } else {
        allSavesOk = false;
      }
    });

    if (!allSavesOk) break;

    // ---------- Save Contract ----------
    group('save-contract', function () {
      const params = {
        headers: {
          'Content-Type': 'application/json',
          'Accept': 'application/json',
          'Object-Type': 'trades-Contract',

          'Transaction-Id': `${trnId}`,
          'Lock-Id': `${lockId}`,
          'Id': `${contractId}`

        params,
        'save-contract'

      const ok = check(res, {
        'save contract: status 200': (r) => r && r.status === 200,

      if (!ok || !res || res.status !== 200) allSavesOk = false;

    group('save-trade', function () {

        `${BASE_URL_TXN}/api/transactions/save`,
        generateCashflow(cashFlowId1, tradeId),
        params,
        'save-cashflow-1'
      );

      const ok = check(res, {
        'save cashflow-1: status 200': (r) => r && r.status === 200,

      if (!ok || !res || res.status !== 200) allSavesOk = false;

    if (!allSavesOk) break;

      const params = {
        headers: {
          'Content-Type': 'application/json',
          'Accept': 'application/json',
          'Object-Type': 'trades-Cashflow',
          'Global-Id': `${globalId}`,
          'Transaction-Id': `${trnId}`,
          'Lock-Id': `${lockId}`,
          'Id': `${cashFlowId2}`,
          'Parent-Id': `${tradeId}`
        },
        tags: { name: 'transaction-service - save' },
      };

      const res = postWithRetry(
        `${BASE_URL_TXN}/api/transactions/save`,
        generateCashflow(cashFlowId2, tradeId),
        params,
        'save-cashflow-2'
      );

      const ok = check(res, {
        'save cashflow-2: status 200': (r) => r && r.status === 200,
      });

      if (!ok || !res || res.status !== 200) allSavesOk = false;
    });

    if (!allSavesOk) break;
  }

  // ------------------------------------------------------------------
  // 5. Commit транзакции (ДО освобождения блокировок)
  // ------------------------------------------------------------------
  group('transaction-commit', function () {
    const params = {
      headers: headers,
      tags: { name: 'transaction-service - commit' },
    };

    const res = getWithRetry(
      `${BASE_URL_TXN}/api/transactions/commit/${trnId}/${processId}`,
      params,
      'transaction-commit'
    );

    check(res, {
      'commit: response received': (r) => r !== null,
      'commit: status 200': (r) => r && r.status === 200,
    });

    if (!res || res.status !== 200) {
      console.error(`FATAL: commit failed for trnId=${trnId}, processId=${processId}`);
    }
  });

  // ------------------------------------------------------------------
  // 6. Release всех блокировок (ПОСЛЕ commit)
  // ------------------------------------------------------------------
  group('lock-release-all', function () {
    for (let j = 0; j < acquiredLocks.length; j++) {
      const lock = acquiredLocks[j];

      const params = {
        headers: headers,
        tags: { name: 'lock-service - release' },
      };
      const payload = JSON.stringify({ lockId: lock.lockId });

      const res = delWithRetry(
        `${BASE_URL_LOCK}/api/locks/${lock.globalId}/release/${lock.lockId}`,
        payload,
        params,
        `lock-release[${j}]`
      );

      check(res, {
        'lock release: status 200': (r) => r && r.status === 200,
      });
    }
  });
}

## Что изменилось по сравнению с предыдущей версией

### Порядок операций теперь такой:

```
1. Batch global-id
2. Batch object-id
3. Begin transaction
4. Цикл по сделкам:
   ├── acquire lock  ──► запоминаем { globalId, lockId } в массив
   ├── save Contract
   ├── save Trade
   ├── save Cashflow 1
   └── save Cashflow 2
   (lock НЕ освобождается!)
5. Commit транзакции
6. Release ВСЕХ блокировок из массива
```

### Ключевые изменения

| Было | Стало |
|---|---|
| `acquire → save × 4 → release` внутри цикла | `acquire → save × 4` внутри цикла, **release вынесен после commit** |
| Блокировки освобождались до commit | Блокировки **накапливаются** в массиве `acquiredLocks[]` и освобождаются **только после успешного commit** |
| При ошибке save блокировки терялись | Даже при `break` из цикла все ранее захваченные блокировки будут освобождены в шаге 6 |
